
    
      
        SErubabel wilt hooren,
        Seer wijs // den prijs // hy heeft,
        Den vroukens uytvercoren
        Haer cracht // met macht // verheeft,
        Oft  den Coninck hooch sweeft,
        Den wijn // seer fijn // om drincken,
        Die wijven // blijven // blincken,
        Boven al datter leeft.
      
      
        Al wat die mans bedrijven,
        Met raet // met daet // met cracht,
        Dat is al voor die wijven,
        Als sy // haer bly // toelacht,
        Al haer arbeyt versacht,
        Veel gout // elck soudt // hooch houwe,
        Een wel // ghestel // de vrouwe,
        Is noch hoogher gheacht.
      
      
        Roemt wat oyt ooghen saghen,
        Wat mensche // wensche // mach,
        Van't aenbegin der daghen,
        Waer deucht // of vreucht // gheschach,
        Van Iubal maeckt ghewach,
        Van al // ick sal // noch hooren,
        Pryeel // casteel // of tooren,
        Sonder vrou ist gheclach.
      
      
        Al int lusthof des Heeren,
        Het welck // noch elck // laudeert,
        Was gheheel gheen verseeren,
        Maer rust // met lust // vermeert,
        Noch was niet contenteert
        Adam // maer quam // in rouwe,
        Tot dat // daer sat // een vrouwe,
        Merckt wat een gave weert.
      
      
        Ras liet Abraham sweeren,
        Sijn knecht // met recht // accoort,
        Wel neerstich informeeren,
        Hem sant // na t'lant // aenhoort
        Daer hy was van gheboort,
        Hy most // gheen cost // daer sparen,
        Sijn soon // dits t'loon // vergaren,
        Een vrouwe van dien oordt.
      
      
        Een vrome vrouw gheneest fris,
        T'verdriet // gheschiet // becans,
        Een vrouwe die God vreest is
        Een schoone // croone s'mans,
        Een gloriose crans,
        Haer reden // zeden // sinnen,
        Verlicht // en sticht // van binnen,
        Isaacx herte gans.
      
      
        Buyten s'lants Iacob diende,
        Den man // Laban // ghetrou,
        Seven jaer doen hy miende,
        Rachel // dit spel // aenschou,
        Te hebben t'eenen vrou,
        Elck jaer // een paer // hy wenden,
        Als daghen // plaghen // t'enden,
        Namt Iacob sonder rou.
      
      
        Esther en Iudith wrochten,
        Gheen man // sulcx can // doen bloot,
        Met haer aenschijn sy brochten,
        Vyant // in schant // ter doot,
        Haer wijsheyt schoonheyt groot,
        Wy lesent // present // beyde,
        Vyant // becant // elck leyde
        So men seyt t'hooft int schoot.
      
      
        Cloeckheyt der vrouwen merckte
        Elck wel // dit spel // ghelooft,
        Debora Barack sterckte
        Iael // seer fel // toeclooft,
        Den Siffera sijn hooft,
        Ick prijs // een wijse // vrouwe,
        Seba // quam dra // in rouwe,
        Abel bleef onberooft.
      
      
        Colstel sijns edele gaven,
        Tot een // sulck reen // u keert,
        Laet haer fonteyn u laven,
        Alleen // en gheen, meer eert,
        Lieffelijck gracieus en weert,
        Vroech spaet // versaet // u wesen,
        Altijt // verblijt // met desen,
        Proverbij vijf t'so leert.
      
      
        Ackers, huysen, wy leset,
        Dat vaer // ons daer // versiet,
        Een vrouwe die God vreeset
        Met tot // ons Godt // om vliedt,
        Door haer // smelt swaer // verdriet,
        Sy gheeft // beleeft // hem cleeren,
        Door sy // comt hy // tot eeren,
        Sonder vrou ist doch niet.
      
      
        Recht heeft die preker schreven,
        Van een // met gheen // ghepaert,
        Wee hem in alle sijn leven,
        Die 't goet // met spoet // vergaert,
        Beter is hy verclaert,
        Een paer // can haer // verwarmen,
        Sijn vrient // verlient // hem d'armen
        So hy struyckelt vervaert.
      
      
        Ach wee is swaer om lesen,
        T'welck hy // brengt by // int perck,
        Wedenaers Weduwen Weesen,
        T'begin // den sin // aenmerck,
        T'is wee t'sy leeck of clerck,
        Een wort // wel cort // vergaende,
        Twee souden // t'houden // staende,
        Een drievout snoer is sterck.
      
      
        Hoort hier na Petrus raden
        Een man // wel can // doort wijf,
        Comen tot Gods ghenaden,
        Die t'woort // verstoort // tot kijf,
        Door der vrouwen bedrijf,
        T'welck niet // gheschiet // met vlechten,
        Noch gout // veel fout // t'aenhechten,
        Om haer uyterlijck lijf.
      
      
        Elck pooch dits mijn vermanich,
        Met vreucht // na deucht // om meest
        Den mannen onderdanich,
        Een sacht // stilach // te gheest,
        Dits hooch voor God gheweest,
        Dus pleghen // deghen // vrouwen,
        Eertijts // den prijs // te houwen,
        Alsmen van Sara leest.
      
      
        Laet haer nu een exempel,
        O wijven // blijven // dijn,
        Al die Gods heylighe tempel,
        Met steenen // meenen // fijn,
        Vercieren als robijn,
        Elck stel // haer wel // te leven,
        V daedt // gheen schaet // laet gheven,
        Om haer dochters te zijn.
      
      
        En ghelijck hem verblijdet,
        Een man // al van // sijn bruyt,
        Esaias belijdet,
        So seer // die Heer besluyt, Sach.2,8
        Ons in liefden beduyt,
        Als t'sin // des min // naers blaket,
        T'aenschou // der vrou // vermaket
        So die Heer ons viertuyt.
      
      
        Niet alleen voor den Vader
        Door deuchd // dees vreuch // de rijst,
        Maer d'enghelen alle gader,
        Van een // onreen // die grijst,
        Want quaet en boet bewijst,
        Lucas // schrijft ras // van desen,
        Vijftien // wilt sien // en lesen,
        Tot hem keeren hy prijst.
      
      
        Daer toe quam Christus goedich
        Verbond // ons won // den seer,
        Als wy onreyn en bloedich,
        Verslaghen // laghen // neer,
        Hy wies, hy salfd ons weer,
        Hy cleed // ons deed // dus croone,
        Sijn trouw // als vrou // we schoone,
        Schanck ons met grooter eer.
      
      
        En souden wy niet hooren
        Den stem // van hem // so soet,
        Comt dochter neycht u ooren,
        Verlaet // u staet // met spoet,
        V volck uws vaders goet
        Doet an // ghespan // en daghen,
        So fel // seer wel, behaghen
        Den Coninck u ootmoet.
      
      
        Roert doch die Schrift van wijven,
        Daer meest // Gods gheest // gaet raen
        Der testamenten drijven,
        Hoe 't oud // noch soud // vergaen,
        Maer t'nieu soud eewich staen,
        Sara // sprack dra // drijft schoone
        Wech gheeren // dienst deeren // haer soone,
        Die en sal gheen erf ontfaen.
      
      
        Wijven zijn d'instrumenten
        Gheweest // self leest // Gods Wet,
        Daer hy sijn Testamenten
        Voorwaer // seer claer // mee set,
        D'eerste was Lea met,
        Ses soon // heel schoon // beschoncken,
        Rachel // seer wel // quam proncken
        Met Ioseph dien hielp bet.
      
      
        Tot verlossing wert sonden
        Den soon // van schoon // Rachel,
        De Sone Gods bevonden
        Een hooft // danckt looft // hem wel,
        Van 'tgheestelijck Israhel,
        Sijn lust // vry blust // in deuchden,
        Sijn wil // houdt stil // in vreuchden
        En leeft na sijn bevel.
      
      
        Het zy mannen of wijven,
        Grieck Iued // voorhued // besneen,
        Die by sijn woort vast blijven,
        Bequaem // zijn tsa // men een,
        Sijn heylighe lichaems leen,
        Sijn lief // sijn brief // sijn tempel,
        Alleen // een reen // exempel
        Aller menschen ghemeen.
      
      
        Merckt oock Aswerus dinghen,
        Datsy // liet hy // ghelijck,
        Door seven camerlinghen,
        Ontbien // om sien // publijck,
        Hoe schoon en suyverlijck,
        Den Heer // sijn eer // te romen,
        Maer stout // sy wout // niet comen
        Is verstooten van haer rijck.
      
      
        Esther t'was een Iodinne,
        In 't goet // ootmoet // bequaem,
        Wert weder Coninginne,
        Ontfinck // alle dinck // daet, naem,
        Dat Vasthi liet met blaem,
        Sy was // dat pas // van staten,
        Niet meer // maer eer // verlaten,
        Maer sy was aenghenaem.
      
      
        Daer staet Remeynen ellef,
        V houdt // niet stout // maer vreest
        Want die natuerlijcke sellef,
        Hoe claer // elck waer // die meest,
        Zijn niet verschoont gheweest,
        Veel schilt // wy wilt // doch waren,
        Ons condt // goet rondt // niet sparen
        Dan door een ander gheest.
      
      
        Een ander gheest te deghen
        Wort claer // Gods schaer // gheleert,
        Van Christo Luca neghen,
        Gheen saeck // met wraeck // verweert,
        s'Gheests vruchten zijn gheeert //
        Veel deucht // als vreucht // en vrede,
        Lanckmoedich // goedich // mede,
        Sulck een Christus begheert.
      
      
        Princesse edele vrouwe,
        't Lams bruyt // comt uyt // verciert
        V bruydegom ghetrouwe,
        Ghelijck // Cantijck // maniert,
        Suyver ongheschoffiert
        Sijn raet // u laet // ghenoeghen,
        Sijn woort // accoort // wilt voeghen,
        En sijnen Sabboth vyert.
