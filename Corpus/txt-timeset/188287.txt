
    
      
        't GRootste vermaecken // dat meest verblijdt
        In quade saecken // is korten tijdt.
        't Meeste verblijden // 't welck 't goede gheeft,
        Sijn lange tijden // dat men het heeft;
        Wanneer ick ter deegh // dit overweegh,
        Denck ick; Geluckigh sijn wy om leegh.
      
      
        't Leven beneden // ons lastigh wordt,
        Maer 't is oock heden // geweldigh kort,
        Schoon wy besueren // hier veel verdriet,
        't Sal niet langh dueren// 't is haest te niet.
        Maer 't leven vol eer // by Godt den Heer
        Dat sal niet eyndighen immermeer.
      
      
        Salig' ellenden // en ongeneught,
        Die altijdt enden // in lust en vreught,
        Saligh vermaecken // dat niet en wordt
        Door eenige saecken // oyt af-gekort,
        En salighen end // dat ons versent,
        In vreught, daer 't eynde gheen vreughde schent.
