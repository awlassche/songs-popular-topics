
    
      
        DAncklijck eenmoedich // prijst Godt den Vaer,
        Want hy seer goedich // is voorwaer,
        Sijn gonst om prysen // gaet hy eenpaer,
        Ons nu bewysen // int openbaer,
        Och vrienden ghy // singht doch met my
        Van herten bly // danckt doch God allegaer.
      
      
        Int huys des Heeren // sullen bereyt,
        Wy gaen met eeren // soo hy toeseyt,
        Om vreuchd' verwinnen // hier boven gaen,
        Ons voeten binnen // de poorten staen,
        Blijft op den ganck // u leven lanck,
        Al zijn wy kranc // die hop' versterct ons saen.
      
      
        Een Stadt ghebouwen // heerlijck en breyt
        Haer Borghers houwen // die eenicheyt,
        Men mach by trouwen // seer hoogh ghestelt
        Haer licht aenschouwen // het lucht int velt:
        Och vrienden jent // maeckt oock bekent,
        Dat ghy nu bent // Borgers elck als een helt.
      
      
        Recht doch u wesen // als Christen goet,
        Aen Ziel ghepresen // hert en ghemoet,
        Als d'eerste deden // met Ziel en lijf,
        Eenich in vreden // te samen stijf,
        Gods glory schoon // spant daer de Croon,
        In 'sHemels Troon // sal blincken haer bedrijf.
      
      
        In dese muyren, eert Godt den Heer,
        Lieft sonder truyren // elckandren meer,
        Gaet opwaerts beden // voor Godt aldaer,
        Drymael 'tgheeft reden // int nieuwe Iaer,
        Hem offer bringht // danckt, looft en singht,
        Dit vyer ontsinckt // de liefde in ons claer.
      
      
        Climt op de graden // volheyt der deucht,
        Dat ghy gheraden // 'tgheloove meucht
        Betuygen waerlijc // als 'tvolck Gods vroet
        Wt Godt schijnbaerlijck // ghesproken soet,
        Met cracht en daet // soo Christus raet,
        Elck huys bestaet // datmen so bouwen doet.
      
      
        Kent, wilt ghevoelen // daer zijn gheset,
        De Rechters stoelen // nae rechter Wet,
        Oordeel te gheven // in Davids huys
        Nae een elcx leven // sonder abuys
        Krijcht elck een slecht // daer loon nae recht,
        D'onnutte knecht // sal daer worden confuys.
      
      
        Perfeckten vrede // soo't Godt behaeght,
        Crijcht niemant mede // dan die nu draeght,
        Liefde te vooren // int herte hier by,
        Reyn uytvercooren // sonder party
        Godtloosen siet // en hebbens niet
        Want vrede biedt // Christus den synen vry.
      
      
        Ick wensch u t'samen // vrede van Godt
        Liefde met namen // 'tnieuwe ghebodt,
        Naer Schrifts verclaren // Ierusalem,
        Wilt openbaren // volght Christi stem,
        In deucht gaet voort // al na sijn woort,
        Niet anders hoort // en sal behaghen hem.
      
      
        En in u toorens // breet ende smal
        Moet zijn te voorens // vred' over al
        Daer men behoedich // de wapens hout
        Bereyt en spoedich // seer menichfout,
        Om de Trompet // te blasen net,
        Zijt ghy gheset // 'tvolck Gods is betrout.
      
      
        Treedt op u voeten // 'tgheluyt nu staet,
        Ontwaken moeten // sy niet te laet,
        Den vred' wilt preken //  straft nu by tijt
        Soo onbesweken // u Ziel bevrijt,
        Gods volck certeyn // de kudde reyn
        Bewaert nu reyn // want ghy haer wachter zijt.
      
      
        En wassen, groeyen // moet ghy Gods stadt
        En vyerich bloeyen // ick wensch u dat,
        Om 'tlof te bringhen // voor Gods aenschijn
        Is nu het singhen // en 'twenschen mijn,
        Vreed'lijck vergaert // Godtlijck gheaert,
        So Schrift verclaert // met Christen sinnen fijn.
      
      
        Recht op't vervallen // bint met der handt,
        Heer, onder allen // des vredens bandt,
        Bouwen, vergaren // souden wy sterck,
        Als trou dienaren // soo aen u Kerck,
        Iae als een man // timmeren dan,
        Gheen vyandt can // ons dan beletten 'twerck.
      
      
        Staet ons verhogen // Heer by met cracht,
        Want onvermoghen // is onse macht,
        Gheensins verblyden // en moghen wy,
        Helpt ghy niet stryden // aen onse zy,
        In u macht staet // ons cracht en daet,
        Verwinnen laet // ons doch met moede bly.
      
      
        Salicheyt nacy // ghy Christen schoon,
        Zy u, soeckt stacy // in 'sHemels throon,
        Dat ghy met namen // sonder beswijck,
        Soo meucht versamen // int Hemelrijck,
        Liefdens vermaen // heeft dit ghedaen,
        Neemt het doch aen // van my so al ghelijck.
