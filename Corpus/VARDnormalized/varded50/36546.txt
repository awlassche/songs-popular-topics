
    
      
        In Jesus name // Broeders eerzaeme,
        Stelt u eenpaar // Nu te beginnen
        Hem te beminnen // Al in dit zoete Nieuwjaar.
      
      
        Een herte reine // Ook anders geyne,
        Dit des al voorwaar // Minnen allene
        het Kindeken kleene // Al in dit zoete Nieuwejaer.
      
      
        Hij heeft geleden // Hij is besneden
        Dit des openbaar // Wilt u besnijden
        Zonder vermijden // Al in dit zoete Nieuwejaer.
      
      
        Zijn heilig sterven // Genoegte derven
        Dits' openbaar // Die doet verwerven
        Dit hemelse erven // Al in dit zoete Nieuwejaer.
      
      
        Och Maria schone // Bid uwen Zone,
        O Maget klaar // Den God hier boven,
        Wij eeuwig loven // Al in dit zoete Nieuwejaer.
      
      
        Vader almagtig // Zone eendragtig,
        Heilig Geest klaar // Een God Dryvuldig
        Verheft ons schuldig // Al in dit zoete Nieuwejaer.
