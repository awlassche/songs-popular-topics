
    
      
        Ik belijt // het met vlijt // op een tijd
        Mij hebben met reen // been,
        Twee verblijd, om een gave cleen,
        Op dat sy't ghebruyckten gemeen,
        Enen sanck // wast begeren,
        Ik dorstet lank // niet consenteren,
        Hierom want // mijn verstand // alle kant
        Gelijke veerdig niet // ziet,
        Nader hand // ik mij noch beriet,
        Ende vant // dit getuige lied.
      
      
        In de leer // die de Heer // zijn volk teer,
        Heeft tot een ghebedt // zet,
        Haar begeer, vervullende net,
        Eyscht hij eer // van elk onbesmet,
        Namelijk dit // zegt te gader,
        Als gij bidt // onze Vader,
        Gij eerzaam // uwe faam // ende naam,
        Gheheylicht alle tijd // zijt,
        Onbekwaam // hen hier in verblijd
        Zonder scham, die gantse wereld wijt.
      
      
        Deze gront // werd vercondt // door die mond
        Van des Heren boon // schoon,
        Te aller stond // ons tot enen loon,
        Om ghesont // krijgen t'eeuwich loon,
        Israhel // Gods lof zongen,
        Doen zij snel // eerst ontspronghen,
        Door die zee // alletwee // twelck haar de
        Farao doen bij gram // kwam,
        Ende mee // haren weg aennam,
        Wijt en bree // gants te grond hij zwam.
      
      
        Tot des leer, // tot Gods eer // noch veel meer
        In het Psalm boek vervaet // staat,
        Alle weer // Davidts gantse raad,
        En begeer // alle hier op staat,
        God te prijzen // groot te maken,
        Te bewijzen // in veel zaken,
        Hoe verheugt // watte vreugd // ende denkt,
        Men in dezen oord // sport,
        Zo hem veucht // hier tegen discoord,
        Die lust der jeugd // dempt met s'heeren woord.
      
      
        Seraphin // Cherubin // ook niet min,
        Haar om zulks te doen // spoen,
        Haren zin // strecket ons ten goen,
        Haar gewin // is al ons behoen,
        Ellick veilig, roept om seere,
        Heilig heilig // heilig Here,
        Onze God // Sebaoth // u gebod,
        V glorieuse raad // gaat,
        Ellick tot // deur en ellick straat,
        Hij is zot // die dit niet en vaat,
      
      
        Een ieder een // groot en cleen // die des reen
        Weg uit zijn gedacht // slacht,
        Wat die meen // is niet en betracht,
        Komt in vreen // als elk Christen lacht
        Hoe gezien // moet dit prijzen,
        Het bedien // zal u wijzen
        Met akkoord // ongestoord // Godes woord,
        Matheus in het vijft // schrijft,
        Deze oord // ons in Christo lift,
        Die t'hoort // ende daar bij blijft.
      
      
        Niet toe laat // dat men kwaad // inder daad
        Van u met waarheid // zeid,
        Altijd staat // haar tot denkt bereid,
        Die u smaad // aandoen ende leit,
        Goed te wensen // haar te stichten,
        Voor die mensen // laat zo lichten
        Vwen ganck // dat ze dank // vrij en vranck,
        Zeggen onzen Heer // zeer
        Met verlanck // volgen uwe leer,
        Sulcken klank // gheschiet God ter eer.
      
      
        Daar toe blij // bidden wij // te allen tij,
        Spreekt Paulus den Heer // zeer,
        Op dat hij // u te recht bekeer,
        Zo dat gij // weerdelijck zijn leer,
        Wandelen zult // al u dagen,
        En vervult // na zijn behagen,
        Met goedheid // dat gezeid // en verbreyt
        Zij door u zijn prijs // wijs,
        Zijn Majesteit // heerlijke logijs,
        Wel bereid // die in deuchden grijs.
      
      
        Enen raad // schreven staat // zelver gaat
        Johannes vijftien // zien,
        Na die maet // moetet recht gezien,
        T'alligaet // zal ik u bedien,
        s'Vaders eer // wordt bedreven,
        Met begeer // als ons leven,
        Dat elk spoed // vruchten doet // overvloed,
        Te worden Christus knecht // recht,
        Hij behoed // in alle gevecht,
        Ons gemoed // hoe simpel of slecht.
      
      
        Al u werk // zwak of sterk // leek of klerk
        Al tot Godes eer // keer,
        In elk perk // vordert zijnen leer,
        Dat gheen zwerk // v duister verneer,
        Zo gij u leven // na t'vleysch wensen,
        Gaat begeven // als andere mensen,
        Ezechiel // waerschout wel // dat gij snel,
        Maakt des Heren naam // blaam,
        V ghesel // daar door komt tot scham,
        Die wereld fel // lastert onbekwaam.
      
      
        Eer ik sluit // dat beduyt // hoort mij uit,
        Maak ik tot u twen // reen,
        Dit geluid // vervult u gebeen,
        Gheen menuyt // of het zij gemeen,
        Mijn begeer // is t'versieren,
        Van Gods leer // in alle manieren
        Deze raad // uwe staat // mede vaat,
        En die een ander dient // mient,
        Paulus laat // Titum zijnen vriend,
        De maet // weten niet verstient.
      
      
        Valse schijn // gheen termijn // latet in,
        V woord of werk raad // daad,
        Al is het u pijn // immers oprecht staat,
        Oftser zijn // oft niet die het aangaat,
        Denkt hier bij // God ik diene,
        Zo zal hij // geduld verliene,
        Gehoorzaam // trouw bekwaam // aangenaam
        V haar die u lont // toont,
        Oftse u naam // lastert ofte hond,
        Dat al tsaem // immers niet verschoont.
      
      
        Mijnen sanck // neemt in dank // elk zo lank
        Dat hij hem verhuert // duert,
        Vwe ganck // onderdanig stuert,
        Dat gheen stank // Gods naam versuert,
        Vwe Heren // ook d'onaerdich,
        Alder eren // achtet waardig
        V verblijd // als gij lijt // in des tijd,
        Om gheen misdaad // kwaad,
        Oftme u smijt // en daar toe versmaad,
        Sulcken strijd // is voor God ghenaet.
      
      
        Even wel // niemand zel // zijn rebel,
        Dien God in den geest //vreest,
        Maar zeer snel // voldoen aldermeest,
        Haar bevel // zelf die Schrift deurleeft,
        Mijnen lied // of mijn praten
        Achtet niet // maar wilt vaten
        Petrus leer // Paulus meer // God ter eer,
        Bij dat elk u schrijft // blijft,
        Mijn begeer // dan genoeg beclijft,
        Looft den Heer // bidt dat hij u stift.
