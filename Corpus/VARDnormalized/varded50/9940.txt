
    
      
        V Vrij verblijd// gij Gods Gemeente,
        Met Christo zijt// gij in vereente,
        Hebt goeden moet// door zijn woord zoet,
        Wie u aan doet// smaad oft vercleente.
      
      
        Het is gewis// die oude Drake
        Genegen is// tot strijdt en wrake,
        Op u verstoort// met fel discoord,
        Maar ziet end' hoort// om wat oorzake.
      
      
        Joannes, in// zijn openbaren,
        Zal u den zin// hier van verklaren,
        T'is om t'bevel// end' Gods instel,
        Zij willende wel// men lietet varen.
      
      
        Oft const gij daar// huichelen mede,
        Gij had met haar// ter wereld vrede,
        Geacht, gh'eert// vrij onverfeert,
        Waar gij, verkeert// in land oft stede.
      
      
        Hier tegen dient// bedocht verstandig,
        Dat s'weerelts vriend// God is vijandig,
        Ende zo wie// God is pardie,
        Och lacy die// moet vergaan schandig.
      
      
        T'is beter God// den Heer aankleven,
        Want zijn gebod// is t'eeuwich leven,
        Hier vrij op acht// zeer wijs bedacht,
        T'volherderns kracht// zal u God geven.
      
      
        Zeer listig kan// Satan// bestrijden,
        T'artijckel van// den Ban// end' t'Mijden,
        Op dat// die stad// waar al een gat,
        Oft den muur plat// aan allen zijden.
      
      
        Als op den berg// zoekt door veel dingen
        Zijn secten erg// hij noch te minghen,
        Met t'Rijck Christi// aanbidt zeid hij,
        Ik zal u vrij// alles ghehinghen.
      
      
        Veel onghesont// bouwen op dezen
        Ijdelen gront// Gods volk geprezen
        Soud' hier in het dal// breed over al,
        Onder t'ghetal// der secten wezen.
      
      
        Maar Belial// hoe sy't gloseren.
        Men niet en zal// gheaccorderen,
        Met Christo fijn// in gheen termijn,
        Zij niet en zijn// te compareren.
      
      
        Princesse rein// gij s'Lams huisvrouwe,
        V Kapitein// blijft vrij getrouwe,
        Dat gij in denkt// verwerven meucht
        D'eeuwighe vreugd// voor korten rouwe.
