
    
      
        VAn enigheid // en rust der zielen mede
        Ben ik bereid // te zingen van den vrede,
        Die daar bestaat // in Christo onzen Here,
        Ons toeverlaat // hem komt toe prijs en ere,
        Hij die ons toont // den weg daar toe te komen,
        Aldaar hij woont // met zijn heyl'ghen en vromen.
      
      
        Hij noot vrientlijck // komt al o gij beladen,
        Neemt snel u wijk // van het gheen u ziel mag schaden,
        Gaat gij maar uit // u zelven wilt verzaken,
        Mijn zoet geluid // zal u troostlijck vermaken,
        Komt maar tot mij // en ik zal u dan tonen,
        Die plaatse blij // daar gij eeuwig zult wonen.
      
      
        Verlaat d'onrust // van s'werelts dominacy,
        Ia gans uytblust // haar ijdel conversacy,
        Haar hoge schijn // haar aanzien zeer bedrieglijk,
        Haar wijn is f'nijn // haar vreugde cord wegh-vlieghlijck,
        Haar schijn rijkdom // is armoedt in het laeste,
        Haar eigendom // ontgaet haar metter haaste.
      
      
        Wilt gij dan een // zijn met dat edel wezen,
        O ziel gaat heen // tot d'eenigheyt geprezen,
        Tot Iesu milt // die bad daar om zijn vader,
        O ziele wilt // letten op u ontlader,
        Hij bad zeer vrij // dat wij mochten een zijne,
        Met hem, als hij // met zijn vader devijne.
      
      
        O ziel aanziet // dit met ogen des herten,
        Al komt verdriet // u vleys aan met veel smerten,
        Gaat maar snel voort // luistert niet na s'vlees raden,
        Dringht door d'engh poort // zoekt volheid der ghenaden,
        Want enigheid // met God compt door afsterven,
        Hieromme scheid // en gij vint s'levens erven.
      
      
        O ziel wilt gij // me rust en vrede vinden,
        Aenhoort doch vrij // hoe zoet spreekt u beminden,
        Neemt mij jok licht // op u wilt willig dragen,
        Keert u gezicht // na mij tot allen dagen,
        Leert dat ik ben // zachtmoedig en ootmoedig,
        Van herten en // ik ben tot helpen spoedig.
      
      
        Ia de fontein // ben ik lieflijk en reine,
        Om u ziel rein // te wassen in dit pleine,
        Dat gij meugt zijn // als Maagden suyverlijcken,
        Om volgen mijn // in als zonder bezwijken,
        Zo vint gij vreed // en rust zeer zoet van binnen,
        O ziele tred // hier toe met hert en zinnen.
      
      
        Mijn jok is zoet // mijn last licht voor de mensen,
        Die in ootmoed // met liefde na mij wensen,
        O ziel wat schat // werd ons uit liefd' geschonken,
        Vliet snel en rat // tot des wijn en wordt dronken,
        Komt mijn Vriendin // eet doch van minder vruchten,
        Laat mij doch in // ik neem weg al u zuchten.
      
      
        O Bruidegom // schoon zijt gij boven maten,
        Ons eigendom // en wilt ons niet verlaten,
        Weest altijd bij // en in ons hert o ene,
        Ons vreed' sijt gij // ons troost en anders gene,
        Ons enig schat // ziet aan u arme schapen,
        Op s'levens pat // leidt ons dat wij niet slapen.
      
      
        Weest wel gerust // mijn schapen uitverkoren,
        Hoogheid maar blust // en al s'werelts verstoren,
        Blijft gij maar klein // ootmoedig en zachtmoedig,
        Met liefde rein // tot barmhertigheid spoedig,
        Verduldelijk // draagt maar u kruis en lijden,
        In mijn schoon rijk // zult gij eeuwig verblijden.
