
    
      
        Maagden blomme, zuiver parceel,
        Die v verheuen // stelt:
        De hooft-somme // gans, en geheel,
        Sydy, daar leun // geld:
        Mij quelt // t'gewelt // der zinnen,
        Van binnen // s'herten gront,
        Want gene
        Zo reene,
        Allene, gij cont
        
          Doleuren
        
        Doen scheuren,
        Voor treuren // terstond.
      
      
        Al dijn vroedheid // goddinnelick,
        Zo hemels af gemaelt,
        Is een zoetheid // gans minnelijk,
        Te vollen nooit verhaalt,
        Mij daalt // en straelt // ter herten
        Veel smert // en // ongeval,
        Achermen!
        Mijn kermen
        Ontfermen // wilt al,
        Of sterven,
        Bederuen,
        Duist-werven // ik zal.
      
      
        Rancxken aardig // moet noot // en // recht,
        Doch zoet van aarde // zijt,
        Mij (onweerdich verstoten // knecht)
        Een sauegaerde // zijt:
        Naar strijd // zendt tijd // van vreden,
        Zijt heden // in mijn kruis
        Zoet // mondig,
        Goed // grondig,
        T'bloet- // hondich // gespuis,
        Laat drijuen,
        Zijn blijuen,
        Catijuen // confuys.
      
      
        In het verzuchten // naar troost // en // raad,
        Ligh ik gans kerker // vast,
        Nu mijn duchten // ten boosten // gaat,
        Den brand noch sterker // wast,
        Dien last // deurtast // mij schandig,
        Vijandig // met veel grief:
        O rare,
        Troostbare,
        In het zware // miskies,
        Van dezen,
        Wilt wezen,
        T'genesen // schoon lief.
      
      
        Als idoone // Princesse // mij
        In het laatste secoureert:
        O mijn schone // Deze // vrij,
        Mijn hulpe mainteneert:
        Cesseert // geleerd // verstandig
        Mijn brandig // lijden straf:
        Of! zonder,
        Groot wonder,
        Ik onder // in het graf
        Moet dalen:
        En halen,
        T'betalen // hier af.
