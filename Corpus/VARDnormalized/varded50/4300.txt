
    
      
        KOR ect // in heusscher zeden,
        Weet ik een vrouwe // fijn:
        Perfect // en zo besneden,
        Mocht noit kersouwe // zijn:
        
          T'sanguijn
        
        Aanschijn,
        Divijn:
        Dunkt mij zo glorieus:
        Mijn herte // gezond
        Deur smerte // gewont,
        Maakt zij victorieus.
      
      
        Notabel // Rethorijcke,
        Haar edel tonge // waeght
        Zo Abel // in muzieke
        En was nooit Jonge // maagd,
        Wie claeght,
        Of knaeght,
        Zij draagt,
        In als haar even // wijs,
        Van klaren // gesanck,
        Of snaren // geklank,
        Moet ik haar geven // prijs.
      
      
        Een // jeugd // is zij bepeerelt,
        Geprezen // wonder // baar:
        Geen // vreugd // in gans de wereld
        Kan wezen // zonder // haar,
        Geen zwaar,
        Gevaar,
        In het iaer,
        Kan mij ter noot // verslaan:
        Bij reden // gezeid,
        Ben heden // bereid,
        Voor haar ter dood // te gaan.
      
      
        Lust draegh ik // in het beminnen,
        Van dezer weelden // vloed:
        Al zag ik // duist goddinnen,
        Versierd als beelden // zoet:
        Lijf, goed,
        En bloed,
        Zijn moet,
        Alleen in haarder // hand:
        Niet blijders // behaagt:
        De nijders // gevraagd,
        Waar is dierbaerder // pand?
      
      
        In vroedheid // zo deurdroncken,
        Kan geen gecroonder // staan:
        Van zoetheid // overschonken,
        Sach ik nooit schoonder // aan,
        Zulk graan
        Werd zaan,
        Misdaen,
        Die haar tooght cleynder // staat:
        Bij crachte // kan // zij
        De clachte // van // mij
        Kronen, met reynder // daad.
      
      
        Als ik // verwonderd // mercke
        Wat zij ter eren // biedt:
        Den strik // van honderd // sterke.
        Vileyns, kan deren // niet,
        Dus vlied
        Verdriet:
        Want ziet
        Mij is den morgen // bij:
        Mijn // zinnen // bezwaard,
        Zijn // binnen // bewaard,
        Ja gans van zorgen // vrij.
