
    
      
        HArderinne// mijn Vriendinne,
        Mijn vermaak en levens vreugd
        Gij allene// zijt de gene,
        Die mijn hert en ziel verheugt.
      
      
        het Vriendlijck wezen//nooit volpresen,
        En u Coralijne mond,
        Waard te noemen// en te roemen;
        Schoonder dan de Morgenstond.
      
      
        Rooder wangen// waard het ontvangen
        Duizend Kusjes, nacht en dag:
        Phoebi stralen// willig dalen,
        Door het lieflijk Oog opslag.
      
      
        Al uwe Leden// net besneden,
        Overschoone Veldt-goddin,
        Doen mij schrijven// ik zal blijven
        Te uwen dienst, in trouwe Min.
