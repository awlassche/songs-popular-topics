
    
      
        Laat ons verblijden // met vrezen en beven
        Bozen schijn mijden // want ons is gegeven
        In onze lijden // in het begin der tijden,
        Macht om te leven.
      
      
        In zond geboren // als de canenijten,
        En gants verloren // ginck met ons versmijten
        Al zoudt wij smoren // niemand kan ons horen
        Priester Levijten.
      
      
        Stramen en wonden // waren al ons leden,
        Dood in der zonden // t'hert was onbesneden,
        Onreine wonden // ongezouten gronden,
        Godloze zeden.
      
      
        Bloot ende naken // lagen wij met schanden,
        In alle zaken // was niet voorhanden,
        Maar God liet maken // rokken wie zal het lakken,
        Van ons vijanden.
      
      
        En ginck ons pronken // met spansels oorringhen,
        Hij heeft ons zonken // sijden kostel dingen,
        Van goud wij bloncken // in het riet gelijk vonken
        Sierlijk wij gingen.
      
      
        Tot alle nasi // is des faam gheresen,
        Van deze gracy // worden wij geprezen,
        Maar wij och lacy // hoe van estimacy,
        Wordt hier bewezen.
      
      
        In die Profeten // leeftment overvloedig,
        Hoe't wordt verweten // Israhel stoutmoedig
        Dat zij vergeten // hoe zij had gezeten,
        Onreyn en bloedig.
      
      
        Alle die dieren // kranen ende swalven,
        Holt zijn manieren zijner komste halven,
        Maar te regieren // mijn volk recht te stieren,
        Helpter gheen zalven.
      
      
        Kermen en klagen // hoe God heeft verheven
        Na zijn behagen // Israhel van leven,
        Maar zij ontsagen // weldaad nochte plagen,
        Tegen te streven.
      
      
        Ossen en muylen // kennen hare Heren,
        Maar ik moet huilen // mijn volk wil niet leren,
        Het brist als guylen // t'graeft malkander kuilen
        Om te verheren.
      
      
        Priesters en Heren // die het volk zouden straffen
        Zijn stom in het leren // honden die niet blaffen
        T'goet zij verkeren t'boose zij eren,
        Terwe moet kaf zijn.
      
      
        Zoudt ik niet wreken // zulke boze daden,
        Ik ginck uytreecken // alle mijn ghenaden,
        Met loose trekken // met hoveerdich spreken
        Sy't al versmaden.
      
      
        Daarom met banden // wil ik ze verzenden,
        Tot haarder schanden // met groter ellenden,
        In vreemde landen, onder haar vijanden,
        Aenschout net enden.
      
      
        Och wat al plagen // honger pestilency,
        Mosten zij dragen // daar toe vyolency,
        Zo vele dagen // dat God mocht behagen,
        Haar penitency.
      
      
        Komt weer uit Babel // nu alle gelijke,
        Acht niet als fabel // Gods woord en zijn rijke
        Werd den diabel // weest niet variabel
        Maar autentijcke.
      
      
        Hier condy sporen // tweederleye leven,
        Dat was verloren // ghinck God eerst verheven,
        Daar na met toren // ginck hijs al verstoren,
        Die kwaad bedreven.
      
      
        Tot alle beiden // lieven ende vrezen,
        Laat u toch leiden // door zingen en lezen,
        Wiltse recht scheiden // u ook recht bereiden
        Tot 'sHemels wezen.
      
      
        Eermen u dringen // wilt van liefde lopen
        Ter bruiloft ringen // wilt gheen ossen kopen
        Noch onschuld bringen // van tijtlijcke dingen
        T'eewich wilt hopen.
      
      
        Recht wordt geprezen // lieven ende minnen,
        Laat zijn geresen // haar vruchten van binnen,
        Zij zal genezen // al uwe vrezen,
        Gij zult verwinnen.
      
      
        Wel hen die het vaten // in een goe conscienty,
        Tijdelijk verlaten // doende penitenty,
        Haar leven haten // t'sal haar vele baten,
        In Gods sententy.
      
      
        Enen Eleaser // zeven zonen tellen,
        Enen Beltsaser // met zijn drie gezellen,
        Nebucadnaser // Antiochus dwaser,
        Woudense vellen.
      
      
        Dood nochte Helle // kan die liefde krenken,
        Stromen rebelle // canse niet verdrenken,
        Al woudy telle // al u goed daar stelle,
        Daar voor toe schenken.
      
      
        Want ellick strijder // worstelt in zijn perken
        Daar zal men ieder // lonen na zijn werken,
        Kwade met lijden, goede met verblijden,
        Och wilt toch merken.
      
      
        Waarom wilt horen // des Apostels raden
        Laat niet verloren // den tijd der ghenaden,
        Of Godes toren // na t'gheclanckx labooren,
        Zoudt u veel schaden.
      
      
        Eer deze dingen // komt die toekomen moete
        Wilt doch geringen // doen oprechte boete,
        Zo suldy zingen // en van vreugde springen,
        Dit is mijn groete.
