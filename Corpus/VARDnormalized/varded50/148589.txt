
    
      
        IN mijnen noot // roep ik tot u, o Here,
        Laat mij niet bloot // want ik ben zwak en tere
        Want ik doch niet // zonder u en vermagh,
        In dit verdriet // troost mij, Heer nagt en dag.
      
      
        Troost mij, o Heer // en wilt mij niet verlaten
        In dit verzeer // komt mij Here te baten,
        Op u alleen // betrouw ik Here mijn,
        Wie mag ons scheen // van uwe liefde fijn?
      
      
        U liefde zoet// bewijst Here geduirigh,
        Wat men ons doet // dat wij niet wezen truirigh,
        Door uwen Geest // Heer, Daniel bij staat,
        In Babel meest // al bij de Leeuwen kwaad.
      
      
        Die Leeuwen fel // zijn ook in de Woestijne
        En zoeken snel // ons te brengen in pijne,
        En slangen fier // spouwen vyerigh fenijn,
        O God zijt hier // onze rechte Medicijn.
      
      
        Geen kruid zo zoet // dat men sou mogen lezen
        Noch plaester goed // en kan ons niet genezen,
        Maar Here gij // die Israel bij-stond,
        Door Christum vrij // maakt ons zielen gezond.
      
      
        Ons zielen krank // zijn altijd in dangiere,
        Door Peste stranck // en sieckt' meniger tiere,
        Zo dat vergat // menig met jammer groot,
        Die haar geen raad, zoeken tegen den dood.
      
      
        Zij minnen meer // dat tijdelijcke leven,
        En des werelds eer // van vleys en bloed gedreven,
        Zij zijn eilaas // zeer jammerlijk verblendt,
        Als Maagden dwaas // door haar traagheid geschend,
      
      
        Zij zijn bij nacht // ook slaperig bedegen,
        Als die geen acht // op het licht hebben gedregen;
        Als haar den Heer // zal opwekken subijt,
        Zullen zij zeer // kloppen, maar ter ontijd.
      
      
        Dan zullen zij, beschaemt, daar buiten blijven,
        Dat roepen vrij // en zal haar niet beklijven,
        het Werd dan te spa // om na Oly te zien,
        Want geen gena // zal haar den Heer verleen.
      
      
        Vrienden gemeen // wilt bidden ende waken
        Omgordt u le'en // laat u licht vyrigh blaken,
        U Zedigheid // laat wezen openbaar;
        Weest zo bereid // als wijze Maagden klaar.
      
      
        Als reine Bruid // wilt u alzo versieren,
        Toont overluid // u licht in all' manieren;
        het Is nu de uir // van den slaap op te staan,
        Door Christum puir // tot den Vader te gaan.
      
      
        Princelijcke God // in eeuwigheid verheven;
        Dit voor een slot // mijn vriendekens gegeven;
        Die van het getal, zijn weer geboren nieuw,
        'K schenk u dit al // voor een vriend'lijck Adieu.
