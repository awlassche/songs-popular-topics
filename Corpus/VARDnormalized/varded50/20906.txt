
    
      
        WEest verblijd // te allen tijd
        des Hemels Koninginne
        Aleluya // alleluya,
        Ik U opdrag all' mijn zinnen.
      
      
        Want Gij hebt, die het all' schept,
        Maagd! in U gedragen,
        Alleluya // alleluya,
        God in U had groot behagen.
      
      
        Dezen Heer // heeft zijn keer
        Uit het graf genomen,
        Alleluya // alleluya,
        En is we'er op de aerd' gekomen.
      
      
        Onverwacht // uit het gracht
        Is uwe Soon verrezen;
        Alleluya // alleluya,
        G'lijck voorzeid Hij had voor dezen.
      
      
        Weest verheugt // weest verneught,
        Waerdste Godes Moeder!
        Alleluya // alleluya,
        Opgestaan is onze Behoeder.
      
      
        Maget bidt // dat wij het Wit
        Van den Hemel rakken,
        Alleluya // alleluya,
        En dat wij ons God genaken.
