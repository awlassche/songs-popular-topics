
    
      
        GEkomen // vernomen
        Is heden in 't stal
        Den Hoeder // den Voeder,
        Den Schepper van all':
        De Knapen // der schapen,
        Verlaten hun Vee;
        En soecken // in doecken
        Blymoedigh Hem dree.
      
      
        Haar keelen // en Veelen
        Die worden gesteld;
        Zij zingen // en springen
        Van vreugden langs veld,
        Zij rasen // en blasen,
        En spelen op 't Riet:
        't Is wonder // dat onder
        Hunlieden geschiet.
      
      
        Zij loopen, zij hopen
        Het Kindtje te zien;
        Zij willen // het stillen,
        Zij willen 't verblien,
        Met kussen // met sussen,
        Met vrindelijckheyd,
        Met kallen // met rallen,
        Met Herders-beleyd.
      
      
        Heus Cloris // en Doris.
        Waar loopt gij dus ras?
        Laat soetjes // uwe voetjes
        Volbrengen haar pas.
        Wij moeten // gaan groeten
        Te samen den Heer;
        't Is reden // 'k wil mede
        Bewijzen Hem eer.
      
      
        O Tyter ,, wat Cyther,
        Wat snaare-gespel,
        Wat Orgel // wat gorgel
        Behouwen wij wel
        Om de ere // des Here
        (Die 't herte verheughdt)
        Te spelen // te quelen
        Met 's herten geneughd.
      
      
        Trompetten // Kornetten
        Behouven wij niet,
        Noch Bullen: // wij zullen
        Van buiten een Lied
        Beginne // uit minne
        Van 't Opperste Goed;
        Waar heenen // onze benen
        Nu loopen met spoed.
      
      
        O Maetjens // ons praetjens
        Die dienen gestaeckt;
        Houdt Herder! // nier varder:
        Het Stal'ken genaakt.
        Laat vallen // ons allen
        Ter aarde ter neêr.
        En groete // dit zoete
        Klein Kindeken teer.
      
      
        Zijt welkoom, zijt welkoom,
        Zijt welkoom weerom
        Bestierder // Vercierder
        Van Sterren // en blom;
        Wien baarde // op de aarde
        Dit Maeghdeken fier,
        Wiens glory // memory
        Wij vieren nu hier.
      
      
        Komt Machten ,, komt Krachten
        Van 't hemelse hof;
        Komt Eng'len ,, vermeng'len
        Uwe stemme tot lof
        Onze Broeders ,, onze Hoeders,
        Met hemelsch' jolijt:
        Zegt Amen: ,, weest te samen
        Met de Herders verblijdt.
      
      
        Messia ,, Maria,
        Wij groeten U te saem;
        Verspreyden // verbreyden
        Wij zullen uwe Naam:
        Hier leggen // onze Weggen,
        Hier laten w' onze Melk;
        Wij jachten // wij trachten
        Na huis toe weêr elk.
