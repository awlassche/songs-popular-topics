
    
      
        Als Palmen groenen // zeer suyverlijc,
        En Cederen doenen // zeer heerlijck,
        Zo moet gij bloeyen // in deugden rijk,
        Wassen en groeyen // zeer lieffelijck,
        Op dat gij o jeugd // in ware deugd,
        Bly en verheugt // altijd recht lichten meugt.
      
      
        * Laat in u schynen // dat ware licht,
        En heel verdwynen // dat u ontsticht,
        Zo moet gij bloeyen // in deugden rijk,
        Wassen en groeyen // zeer lieffelijck,
        Op dat gij o jeugd // in ware deugd,
        Bly en verheugt // altijd recht lichten meugt.
      
      
        Stelt u ghelaten // in stilligheyt,
        En wilt aenvaten // goedwilligheid,
        Zo zal dan op dijn // vloeyende zijn,
        Godes Geest fijn // alst water uit den Rijn.
        Op dat gij o jeugd // in ware deugd,
        Bly en verheugt // altijd recht lichten meugt.
      
      
        * Zo zal dan vloeien // van u eerbaer,
        Stromen die vloeien // zeid die schrift claer,
        Hoe zijnen gave // heeft die ontfaen,
        Geen rijcker grave // en vintmen saen,
        Dan die mens rein // ootmoedig cleyn,
        Die zo alleyn // drinkt uit ‘s levens fonteyn.
      
      
        * O reine ziele // die zo gheplant
        Staat in priele // schoon en playsant,
        Wel rijk gesegent // zijt gij voorwaar,
        Want op u reghent // voor ende naar,
        Van God zeer goed // uit genae zoet,
        Des gheestes vloet // waar voor gij wassen moet.
      
      
        * O jeugd bekwame, in sulcken hof
        Wens ik u t’same // om daar Gods lof
        Recht te verbreyden // als spruiten jent,
        Vol vruchtbaerheyden // zeer zijn ge-ent,
        Aan die wijnranck // vol zoeten drank,
        Daar niet uit spranck // dan God tot lof en danck.
