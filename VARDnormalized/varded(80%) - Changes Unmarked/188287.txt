
    
      
        't GRootste vermaecken // dat meest verblijdt
        In kwade zaken // is korten tijd.
        't Meeste verblijden // 't welk 't goede gheeft,
        Zijn lange tijden // dat men het heeft;
        Wanneer ik ter deeg // dit overweegh,
        Denck ik; Gelukkig zijn wij om leegh.
      
      
        't Leven beneden // ons lastigh wordt,
        Maar 't is ook heden // geweldig kort,
        Schoon wij besueren // hier veel verdriet,
        't Zal niet langh dueren// 't is haast te niet.
        Maar 't leven vol eer // bij God den Heer
        Dat zal niet eyndighen immermeer.
      
      
        Salig' ellenden // en ongeneught,
        Die altijdt enden // in lust en vreugd,
        Saligh vermaecken // dat niet en wordt
        Door enige zaken // oyt af-gekort,
        En salighen end // dat ons versent,
        In vreugd, daar 't einde gheen vreugde schent.
