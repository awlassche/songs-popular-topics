
    
      
        Ik arme Kind // waar zal ik keren
        Dan mij verneeren // onder Gods hand,
        Dat mij den wind // der valscher leren
        Niet en magh deeren // Heer doet bystant.
      
      
        O Heer God Vader // vol van genade,
        Van alle quaede // mij doch bewaard,
        Gij Fonteyn-ader // vloeyt vroeg en spade
        Op dijnen zade // van dijner aard.
      
      
        O God, doet mij // verjubileeren,
        Ik zal landeeren // u Name groot,
        Al soud' men vrij // mij persequeren
        En tribuleeren // tot in der dood.
      
      
        Zal ik niet haast // met u verblijden,
        In witter zijden // o Vader jent?
        'T leven verbaasd // is niet dan strijden
        Tot allen tijden // druck en ellendt.
      
      
        Tot u, o Here // wil ik mij geven,
        Mijn steen verheven // mijn toeverlaet,
        Want in u leer // vind' ik het leven,
        Hoe soud' ik beven // als gij bystaet.
      
      
        Ik bidd' u seer // om uwe gracy,
        Tot elker spacy // verhoord u Bruydt,
        Vertroost u teere // door jubilacy,
        Als 's Werelts nacy // zal roepen luydt.
      
      
        Als haar de Mensen // zeer verblijden
        Iagen en rijden // nae's Werelts loop,
        Dan is mijn wenschen // en mijn belijden,
        Wilt mij bevrijden // voor sulcken hoop.
      
      
        Na u allene // zijn mijne gangen,
        Ik ben behangen // met zonden snoot,
        Na anders gene // staat mijn verlangen,
        Wilt mij ontvangen // Heer in der noot.
      
      
        Zal does turbacy // noch lange duiren?
        Dit doet mij truiren // troost mij, o Heer,
        Want door blamacy // zij veel vervuiren,
        Door het versuiren // in uwe leer.
      
      
        Zo wilt bewaeren // sparen en wachten
        Mij door u krachten // mijn Middelaer,
        Wilt mij vergaren // leiden en achten,
        Bij de geslachten // onder 't Outaer.
      
      
        O God mijn Heerder, doet mij doch groeyen
        Door liefde bloeyen // als rank vruchtbaar,
        Ik weet geen weerder // wilt mij besproeyen,
        Laat op mij vloeien // u Geest eerbaer.
      
      
        O gij Fonteine // vol goeds geprezen,
        Daar uit geresen // is't leven jent,
        Gij kont alleyne // ons Ziel genesen,
        Wilt bij ons wezen // tot in den ent.
      
      
        Nu bid' ik Prinse // mijn God almachtig
        Weest nu doch achtigh // u Schaepkens al,
        O Heer, bemintse // uit liefden krachtig,
        Weest ons gedachtig // in't jammer-dal.
