
    
      
        Gebenedyt // menichfout is de man
        Die hier recht doet, in d'Armen noot t'aenschouwen,
        En God altijd// vreest, Zijn Wett' houden can,
        In over-vloet// komt hem den Heer bedouwen:
        Wie in't benouwen// zijn Naasten biedt de hand,
        t'Sal ja// hier na// hem niet mogen berouwen,
        Wanneer// de Heer// slaets eer// der bozen standt.
      
      
        De tijd is cleyn// dat den Godlosen bloeyt,
        Zo werden zij// nergens niet meer gevonden,
        Maar int gemeyn// den vromen int land groeydt,
        God zegent vrij// haar zaad tot allen stonden:
        Die uit liefts gronden// den Armen deucht bewijst,
        Voorspoet// haar moet// die t'geen hem God gezonden
        Hier heeft// mild' geeft// vroom leeft// en Gods Leen spijst.
      
      
        Salich-geluck// werd haar van God belooft,
        Eewich te zijn// met hem in vreugd verheven,
        Die hier in druck// d'Arme die troost behooft,
        Als medecijn// barmhertich komt beneven:
        Die mild'lijck geven// leenen en jonstich bien,
        Uytset// geen fret// noch winst en eyscht int leven,
        V stilt// God wilt// weer milt// rijcklijck verzien.
      
      
        En keert doch niet// u Ogen van den noot
        Der Armen hier// maar wilt wel op haar achten,
        Christus gebiet// zulks, al waar hy't zelfs bloot,
        Zo goedertier// haar te ontvangen, te wachten,
        Haar smert versachten// niet uit dwanck der Gheboon,
        God koos// altoos// d'Arme tot Erff-geslachten,
        Wie claeght// jonst draagt// t'behaeght// God die ist loon.
      
      
        Wie hem ontfarmt// des Armen in ellendt,
        Leent God t'welck wind// dat den Prinsch hooghst geprezen
        V weer erbarmt// als een Moeder haar wendt
        Tot haar zelfs kint// met liefde te genesen:
        Wilt vriend'lijck wezen// uwen Naasten gerieft,
        Onthoudt// van d'Oud'// vaders, alzo wy't lezen,
        Die't deen// Gods leen// wilt cleen// en Aanziet Lieft.
