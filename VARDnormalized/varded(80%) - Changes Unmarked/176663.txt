
    
      
        In mijnen geest  //  moet ik verjubileeren
        Nooit zoete feest  //  dan den Sabboth des Heren
        Wie hem met lust  //  viert van zijn kwade werken
        Van zonden rust  //  en God die hem kan sterken
        Die heeft zeer pleyn  //  den Sabboth recht gehouden,
        God heeft hem rein  //  zijn zonden kwijt gheschouden
      
      
        Ook zalig dan  //  is hij tot allen stonden
        Al sulcken man  //  en rekent God geen zonden
        In wiens gheest fris  //  gheen bedroch noch quae perten
        Gevonden is  //  ook niet dobbel van herten
        Maar als Gods knegt  //  de vrome Iob geprezen
        Slecht ende rechtvolherdich in Gods vresen
      
      
        Ook die gheen ghelt  //  op woeker heeft ghegheven
        Noch geen geweld  //  aan iemand heeft bedreven
        Maar leent met vlijt  //  den armen t’elcker ure
        Zonder respijt  //  alleen uit liefden puyre,
        En geeft haar al  //  die hem niet konnen loonen
        Dien sulcken zal  //  in Godes Tempel wonen
      
      
        Zo wie met vliet  //  uit lust hier beklijven,
        Zij zullen niet  //  ledig onvruchtbaar blijven
        In kentenis  //  van Gods volmaakte leer,
        Die heylsaem is  //  in Christo onzen Here
        God die vermeert  //  de gaef zijns geest beseven
        Wie hen verneert  //  die wordt van God v’heven
      
      
        Is een in noot  //  die hem dan troost van herten,
        En breeckt zijn brood  //  een ziel vol honghers smerten,
        Die niet en kan  //  rusten eten noch drinken
        Als hij een man  //  nootdruftig kan bedincken
        Maar hem eerst spijst  //  uit liefde na en verre
        Zijn licht dat rijst  //  op als de morgen-sterre.
      
      
        Als hij bezorgt  //  de Weeskens rein van zede
        Geeft, leent, en borcht  //  elk na geleghenthede,
        Die haar zo kan  //  voorstaen in alle zaken,
        Regt als haar man  //  raad tot bidden en waken
        Geestigh ghesint  //  sacht, stil tot allen tijden
        Dat zij verwint  //  haar druk en ‘s vijands strijden
      
      
        Niemand en kan  //  tot dezen Sabboth komen,
        Oft ten zij dan  //  dat hij tot sijnder vromen,
        Gantschlijck versaeckt  //  die wereld met haar lusten,
        En niet en smaekt  //  dat aersche vol onrusten,
        Met zijn gemoet  //  gants draegt in God verheven,
        Iae suyr en zoet  //  danckbaerlijcken te leven.
      
      
        Zeer onbesmet  //  is voor God eerweerdig,
        Die niet en zit  //  bij de spotters lichtveerdig,
        In haren raad  //  nog in haar kromme wegen
        Geenszins en gaat  //  maar contrary daar tegen,
        Al zijnen ganck  //  heeft in het Hemelsch wezen,
        Met lof en danck  //  als een met God verresen.
      
      
        Zoet ende fijn // moet ‘t eynd’ aller geboden
        De Sabboth zijn  //  daar maar een is van noden.
        Dits ‘t eenig een  //  stil houden God gelaten
        Als Magdaleen  //  die koos tot haarder baten,
        ‘t Best eenig goed  //  ‘t welk haar bij is ghebleven.
        D’oprechte vroet  //  en is geen wet ghegheven
      
      
        O Gods Gheest net  //  gij kond ons wel regheren
        Gij zijt de wet  //  niemand heeft u te leren
        God heeft u voort  //  den heylighen gegheven,
        En na zijn woord?  //  in haar harten geschreven,
        Gij zijt een licht  //  lanteerne t’haerder voeten
        Na Davids dicht  //  ‘t welk zij na volghen moeten
      
      
        Och liefden graet  //  gij bant aller secreten
        In u bestaat  //  de Wet en de Profeten
        Gij zijt niet smal  //  het eynd’ end’ de hoofd somme,
        Der wetten al  //  haar oorsaeck en waaromme
        Na Schrifts uytlegh  //  groot boven alle gaven,
        De hoogste weg  //  de rust doch aller slaven
      
      
        Nu Prinse God  //  diet alles werkt in allen
        Gij zijt mijn lot  //  laat mij van u niet vallen,
        Werkt nach en dach  //  wilt al mijn kracht regieren,
        Op dat ik mach  //  in u den Sabboth vieren,
        U jok is zoet  //  u last is licht om draghen,
        Wie leert ootmoet  //  zijn ziel rust te allen dagen.
