
    
      
        GRoot wonder // nu onder den hemel gheschiet.
        Een maghet // die draghet een kindeken ziet,
        Te voren // geboren: nooit mensch sulckx sach,
        Dat een suyver maghet van kind ghelach.
      
      
        Hoort lieden // 't gheschieden, dat God heeft behaagt,
        Te geven // het leven door die suyver Maagd.
        Te voren // verloren die waren als dood,
        Die kwam hij verlossen door liefde zeer groot.
      
      
        Veel suerheyt // en stuerheyt des winters men sach:
        Als jeuchdich // en vreuchdich does Maghet ghelach.
        Zij baarden // op aarden den Schepper van al
        In enen zeer cleynen door-luchtighen stal.
      
      
        Doen waren // daar scharen van herder ghestelt,
        Die wachten // bij nachten de schaepkens op 't veld.
        Zij hoorden // de woorden van d'Enghelkens zoet:
        Loopt Herder, loopt vlijtich na Bethlem met spoet.
      
      
        Zij liepen // en riepen malcanderen aan:
        En quaemen // te saemen na Bethlem gegaan.
        Zij vonden // bewonden een kindeken cleen
        Met luyers en tuyers bij Moeder alleen.
      
      
        Doen vielen, door knielen de Herderkens neer:
        Ontladen // aen-baden dit kindeken teer.
        Zij lachten // en dachten: wat wonderen nacht,
        Die ons een zo zoete kint voort heeft gebracht.
      
      
        Doen voeyden // en hoeyden den Engel haar vee.
        De schapkens // al rapkens door-spronghen de wee.
        De geiten // in 't bleyten die schenen verheugt:
        De koeien // in 't loeyen die schenen vol vreugd.
      
      
        O mensen // die wenschen naar dit droeve dal
        Te loven // hier boven den Stierder van al.
        Zoekt vlietich // en schlieich dit hemelse kint,
        'T gheeft deuchden // en vreuchden aan die het bemint.
