
    
      
        GOdlijck-vercierde // Vrouw,
        Die vierig in mijn sinnen // staat,
        Vw’ hoogh-gevierde // trouw
        Mee-doogend’, mij gewinnen // laat,
        Op hem ’t beminnen // slaat,
        Die door dijn leven // leeft,
        En ’t zijn begeven // heeft.
      
      
        Dijn leven leeft // in mij,
        En heeft het mijn gedreven // neer,
        Mijn leven sweeft // in dy,
        En zoekt zijn s’levens leven // weer,
        Daar u verheven // eer
        Mijn trouwe deughden // loont,
        En ’t hert met vreugden // kroont.
      
      
        Verflaude krachten // rijst,
        En lof-sanght u verkoren // Bruyt,
        Die uwe gedachten // spijst,
        Met vreugt, en haren toren // sluit:
        O wel-geboren // spruyt!
        Dijn weerdt-gepresen // aard
        Mijn ziels genesen // baert.
      
      
        Den schoon gehulden // Krans
        V voor-hooft heeft om-dreven // wis,
        Die vol van gulden // glans
        Der Goden, zelfs door-weven // is,
        Door u gegeven // is
        Den geur van Floras,, schat,
        Met ’t eel Pandoras,, vat.
      
      
        Den nijd zijn herte // knaeght,
        Om dat uwe deugd victory // vecht,
        Daar zij haar smerte // klaeght,
        Gij tekens der gelory // recht,
        ’t Welk in memory // hecht
        Des tijdts geweteniss’,
        Vrij van vergeteniss’.
      
      
        Princes, die boven // all’
        Wat aerts is, uwen throone // plant,
        Daar elk u loven // zal,
        Dat gij der maegden krone // spant:
        Jont mij uwe schone // hand
        Die mij voor ’t sneven // hoed,
        En stervend’, leven,, doet.
