
    
      
        V Vrij verblijdt // gij Gods Gemeente,
        Met Christo zijt // gij in vereente,
        Hebt goeden moet // door zijn Woord zoet,
        Wie u aendoet // smaet oft verkleente.
      
      
        Het is ghewis // die oude Drake
        Gheneghen is // tot strijdt en wrake,
        Op u verstoort // met fel discoort,
        Maar ziet end' hoort // om wat oorzake.
      
      
        Ioannes in // zijn openbaren,
        Zal u den sin // hier van verklaren,
        Tis om 'tbevel // en Gods instel,
        Zij wilden wel // men lietet varen
      
      
        Oft kond gij daar // huychelen mede,
        Gij had met haar // ter Wereld vrede,
        Gheacht, gheeert // vrij onverseert,
        Waar gij verkeert // in land oft stede.
      
      
        Hier tegen dient // bedocht verstandig
        Dat swerelts vriend // God is vyandigh,
        Ende zo wie // God is partie,
        Och lacy die // moet vergaen schandigh.
      
      
        T'is beter God, den Heer aankleven,
        Want zijn ghebodt // is t'eeuwigh leven,
        Hier vrij op acht // zeer wijs bedacht,
        'tVolherdens kracht // zal u God geven.
      
      
        Zeer listigh can // Sathan // bestrijden,
        'tArtijckel van // den Ban // en 'tMijden,
        Op dat, de stad, waar al een gat,
        Oft den muur plat, aan allen zijden.
      
      
        Als op den berg // soect door veel dingen
        Zijn secten ergh // hij noch te minghen,
        Met 't Rijk Christi // aanbidt zeid hij,
        Ik zal u vrij // alles ghehinghen.
      
      
        Veel onghesont // bouwen op dezen
        Ydelen gront // Gods volk ghepresen
        Soud' hier int dal // vreedt over al,
        Onder 't ghetal // der secten wezen.
      
      
        Maar Belial // hoe sy't gloseren,
        Men niet en zal // gheaccorderen,
        Met Christo fijn // in gheen termijn,
        Zij niet en zijn // te compareren.
      
      
        Princess rein // gij 'sLams huisvrouwe,
        V Capiteyn // blijft vrij ghetrouwe,
        Dat gij in deucht, verwerven meucht
        D'eeuwighe vreugd, voor korten rouwe.
