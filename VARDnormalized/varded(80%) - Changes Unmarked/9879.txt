
    
      
        WAnneer men qua gheruchten // hoort,
        Van vluchten // voort,
        Ist dan al veel met zuchten // 'twoort
        En duchten // stoort
        Het goed // ghemoet // van binnen // maar
        Te vroeg zijn ons sinnen // zwaar,
        Want 'tkan wel wezen,
        Dat niet zo groot // en is den noot,
        Als vrees in dezen.
      
      
        Die qua maren beseven // zij,
        Doen beven // mij,
        Sechtmen, waar zal men leven // bij,
        Met weven // wij
        En spinnen // winnen // quaelijck // ghelt,
        Dat ons principalijck // quaelt,
        Dus int ghemeene,
        Is grote diert' // onghemaniert,
        En neeringh kleene.
      
      
        Doch neeringhe van lijnen // manck
        In synen // ganck,
        Can somtijdts om verdwynen // lanck,
        Wel schynen // krank,
        Maar saen // vergaen // noch sterven // ziet,
        Men mach 't lijnen derven // niet,
        In 'swerelts ronde,
        Ten waar dat men // noch kin noch ken,
        Daar in en vonde.
      
      
        Voor al in dit verseeren // met
        Des Heren // Wet,
        Moeten wij lijdtsaem leren // bet,
        Afkeeren // het
        Al 't onverduldich wroeghen // fel,
        Laten ons ghenoeghen // wel,
        Met dat voorhanden
        Is, hoe benouwt // die God betrouwt,
        Komt niet tot schanden.
      
      
        Laat hopen om ghenesen // zoet,
        Op dezen // voet,
        Ist kwaad, het kan haast wezen // goed,
        Gheresen // spoet,
        Ghelijck den tijd zijn ganghen // gaat,
        Wij moeten ontvangen // kwaad
        In dancke mede,
        Ghylijck als Iob // zijn antwoord' op
        'tWijfs reden dede.
      
      
        Prinsen die met den kwaden // tyt
        Beladen // zijt,
        Laat u Gods Woord beraden // wijt,
        De schaden // kwijt,
        Werd gij, ja al 't bezwaren // van
        De fell quaede maren // dan,
        Die somtijts rijsen,
        Niet is zo goed // dan wat God doet,
        Van als hem prijzen.
