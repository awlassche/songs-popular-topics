
    
      
        T'Is nu den tijd // subijt // zoet aenghenaem,
        Die Christen zijt // verblijdt // u al te saem,
        En acht gheen blaem, laster noch schande,
        Schimp en verwijt // maar lijdt // door alle faem
        Loopt tot den strijdt // int krijt // maect u bequaem,
        End' u lichaam // offert te pande,
        Ontsteken met der liefden brande,
        Op den Outaer // voorwaar // Christus zeer
        Tot een levende offerande,
        Zo krijcht gij daar // eenpaer // in lijden zwaar,
        Gods Gheest troostbaer // ten onderstande.
      
      
        Van 'sHeeren Geest // den keest // der liefden smaakt,
        Door geen tempeest // bevreesd // niet en versaect
        Volstandigh blaeckt, zonder verflouwen,
        Hout blyde feest // nu eest // bidt ende waect
        In dit foreest // u meest // in deuchden maect,
        Zorgt niet, al naakt // u veel benouwen,
        Wilt Gods beloften vast betrouwen,
        Wat dat gij doet // wilt vroet // om u voorspoet,
        Christum dijn voorbeeldt wel aenschouwen
        Lijdt gij om goed // hebt moet // met trooste zoet,
        Zal u den bloed // des Gheests bedouwen.
      
      
        'sGeests vloet versacht // met kracht, des lijdens grief,
        Gij Gods geslacht // en acht // op geen misschief,
        Hebt altijts lief // die komst des Heren,
        Die onbedacht // in nacht // werd als een dief
        Dus in eendracht // verwacht // die u verhief
        Wt Paulus brief // daar meugt gij leren
        Hoe dat alle dit Aerdtsch verseeren,
        Pijn en verdriet // al niet // is weerdig iet,
        Des Hemels vreugd niet om vermeeren,
        Dus niet en vliet // als riet // Loths wijf aanziet,
        Wat haar gheschiet // doort ommekeeren,
      
      
        Een soutcolom, om t'omsien werd zij saen
        Volgt na Iesum // Christum // zijn baen wilt gaan,
        Zonder inslaen // t'eenigher zijden,
        Maakt geen verschrom // wilt vrom // ghewapent staan:
        Hebt t'eener som // ooc rom // int cruys gelaen
        Wilt onderdaen // Wettelijck strijden,
        Gij zult ontfaen t ghebenedijden,
        Een excellent // en jent // schoon croonement,
        Want die volstandigh zonder mijden
        Is tot den endt // bekent // zal ongheschendt,
        Christum ontrent // eeuwig verblyden.
