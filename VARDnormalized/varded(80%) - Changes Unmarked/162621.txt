
    
      
        VAn eenicheyt // en rust der zielen mede
        Ben ik bereid // te singhen van den vrede,
        Die daar bestaat // in Christo onzen Here,
        Ons toeverlaet // hem komt toe prijs en ere,
        Hij die ons toont // den weg daar toe te comen,
        Aldaar hij woont // met zijn heyl'ghen en vromen.
      
      
        Hij noot vrientlijck // komt al o gij beladen,
        Neemt snel u wijck // van't gheen u ziel mach schaden,
        Gaat gij maar uit // u zelven wilt versaken,
        Mijn zoet gheluyt // zal u troostlijck vermaken,
        Komt maar tot mij // en ik zal u dan tonen,
        Die plaatse bly // daar gij eeuwich zult wonen.
      
      
        Verlaat d'onrust // van s'werelts dominacy,
        Ia gans uytblust // haar ydel conversacy,
        Haar hooghe schijn // haar aanzien zeer bedrieghlijck,
        Haar wijn is f'nijn // haar vreugde court wegh-vlieghlijck,
        Haar schijn rijkdom // is armoedt in het laeste,
        Haar eyghendom // ontgaet haar metter haaste.
      
      
        Wilt gij dan een // zijn met dat edel wezen,
        O ziel gaat heen // tot d'eenigheyt geprezen,
        Tot Iesu milt // die bad daar om zijn vader,
        O ziele wilt // letten op u ontlader,
        Hij bad zeer vrij // dat wij mochten een zijne,
        Met hem, als hij // met zijn vader devijne.
      
      
        O ziel aanziet // dit met ogen des herten,
        Al komt verdriet // u vleys aan met veel smerten,
        Gaat maar snel voort // luistert niet na s'vlees raden,
        Dringht door d'engh poort // zoekt volheid der ghenaden,
        Want eenigheyt // met God compt door afsterven,
        Hieromme scheyt // en gij vint s'levens erven.
      
      
        O ziel wilt gij // me rust en vrede vinden,
        Aenhoort doch vrij // hoe zoet spreekt u beminden,
        Neemt mij jok licht // op u wilt willich draghen,
        Keert u ghesicht // na mij tot allen dagen,
        Leert dat ik ben // sachtmoedigh en ootmoedigh,
        Van herten en // ik ben tot helpen spoedigh.
      
      
        Ia de fonteyn // ben ik lieflijck en reine,
        Om u ziel rein // te wassen in dit pleine,
        Dat gij meugt zijn // als Maagden suyverlijcken,
        Om volgen mijn // in als zonder bezwijcken,
        Zo vint gij vreed // en rust zeer zoet van binnen,
        O ziele treed // hier toe met hert en sinnen.
      
      
        Mijn jok is zoet // mijn last licht voor de mensen,
        Die in ootmoet // met liefde na mij wenschen,
        O ziel wat schat // werd ons uit liefd' geschonken,
        Vliet snel en rat // tot does wijn en wordt dronken,
        Komt mijn Vriendin // eet doch van mijnder vruchten,
        Laat mij doch in // ik neem weg al u zuchten.
      
      
        O Bruydegom // schoon zijt gij boven maten,
        Ons eyghendom // en wilt ons niet verlaten,
        Weest altijd bij // en in ons hert o ene,
        Ons vreed' sijt gij // ons troost en anders gheene,
        Ons enig schat // ziet aan u arme schapen,
        Op s'levens pat // leidt ons dat wij niet slapen.
      
      
        Weest wel gherust // mijn schapen uytvercoren,
        Hoogheid maar blust // en al s'werelts verstoren,
        Blijft gij maar cleyn // ootmoedigh en sachtmoedigh,
        Met liefde rein // tot barmhertigheid spoedigh,
        Verduldelijck // draagt maar u cruys en lijden,
        In mijn schoon rijk // zult gij eeuwig verblijden.
