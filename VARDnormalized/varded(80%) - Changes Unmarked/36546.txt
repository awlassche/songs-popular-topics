
    
      
        In Jesus name // Broeders eerzaeme,
        Stelt u eenpaer // Nu te beginnen
        Hem te beminnen // Al in dit zoete Nieuwjaer.
      
      
        Een herte reine // Ook anders geyne,
        Dit 's al voorwaar // Minnen allene
        't Kindeken kleene // Al in dit zoete Nieuwejaer.
      
      
        Hij heeft geleden // Hij is besneden
        Dit 's openbaer // Wilt u besnyden
        Zonder vermyden // Al in dit zoete Nieuwejaer.
      
      
        Zyn heylig sterven // Genoegte derven
        Dits' openbaer // Die doet verwerven
        Dit hemelse erven // Al in dit zoete Nieuwejaer.
      
      
        Och Maria schone // Bid uwen Soon,
        O Maget klaer // Den God hier boven,
        Wij eeuwig loven // Al in dit zoete Nieuwejaer.
      
      
        Vader almagtig // Zone eendragtig,
        Heylig Geest klaer // Een God Dryvuldig
        Verheft ons schuldig // Al in dit zoete Nieuwejaer.
