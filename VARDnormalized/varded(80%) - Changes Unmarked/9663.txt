        Jonkvrouwe // klaer ghepresen,
        O Syon volgt u Vorst getrouwe // naar,
        Int nouwe // maar, na dezen
        Zal wezen // u beloonen // wijdt,
        Als gij int eeuwig kroonen // zijt,
        Den schoonen // tijd, des Meys deelachtig
        In volkomen wezen krachtig.
      
      
        Voorbij is al des winters tegen // stoot,
        Den reghen // groot, verdreven,
        T'veldt staat gebloeyt in aller wegen // root
        die pleghen // dood, zijn leven,
        Het geven // groene rijsen // klanck,
        Den Mey-boom groen // tot prijs en // dance
        Met wijzen // sank, en nieuwe tonghen,
        Wordt verwinninghe ghesonghen.
      
      
        Syon verblijd in dit verwinnen // want
        Tis binnen // tlandt, al vrede,
        De liefd' uws vriends in uwe sinnen // plant
        In minnen // brand, hij mede,
        De stede // is hij bereyen // gaan,
        Daar alzo schone Meyen // staan,
        V schreyen // saen, stelt al bezijden,
        In den Mey wilt u verblijden.
      
      
        Princess Bruydt, met enen stijven // moet,
        Laat drijven // vloet, van vreuchden,
        In uwen gheest, des Meys inlijven // zoet,
        Beklijven // doet, in jeuchgden,
        Zijn deuchden // zult gij romen // al,
        Daar gheenen winter komen, zal,
        Tot blomen // val, noch heet verbroeyen,
        En gij zult daar eewigh bloeyen.
