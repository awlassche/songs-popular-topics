
    
      
        CEderboom jeughdich groen // plantsoen //
        Deurluchtich paragoen // u doen //
        Is zo betrout //
        Waar ik ter dood benout //
        Gij sout // zijn mijn rantsoen.
      
      
        Abel, en magnifijck // gelijk
        Pallas, die gans aardrijk // publijck
        Sticht int gemeen,
        Zijt gij beelde alleen,
        Want geen // zo autentijck.
      
      
        Troostbaer in tegenspoet // onsoet
        Sydy, als t'quaet gebroet // mij doet
        Zuchten vervaert:
        Bij dien vreugd openbaart
        T'beswaert // treurich gemoet.
      
      
        Hierom ben ik perfect // gewect,
        Een dienaar t'sijn subject // gestrect,
        Voor u: zo mach
        Einden (naar langh verdrach)
        T'geclach // secreet bedect.
      
      
        Als gij (Liefste) dit wrocht // mij docht
        Gij waert zo wtgesocht // ten mocht
        Schoonder Robijn
        Onder der zonnen schijn,
        Niet zijn // zo dier gecocht.
      
      
        Rond om d'aerdtsche tonneel // geheel
        En is liever Juweel // tot deel
        Gewenscht voorwaar,
        Dan u, die zijt een zwaar
        Dierbaer // gulden parceel.
      
      
        In mijn hert nooit gegront // en stond
        Zulk lief, wiens rooden mond // mij iont
        Troostich geniet,
        In alle mijn verdriet,
        Aanziet // gij mij gewont.
      
      
        Nu gij mij diligent // bekent,
        In dienst, geen tijd absent // gewend,
        Schinct mij (voor loon)
        Wt uwen hogen troon
        Een schoon // coronnement.
      
      
        Aangenaam edel kint // gesint,
        Naar dien gij u dus vint // bemint,
        Jont uwen knecht
        Bystant, dat hij met recht,
        T'gevecht // eens overwint.
