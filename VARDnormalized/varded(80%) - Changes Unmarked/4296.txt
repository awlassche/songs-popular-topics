
    
      
        Maagden blomme, suyver parceel,
        Die v verheuen // stelt:
        De hooft-somme // gans, en geheel,
        Sydy, daar leven // geld:
        Mij quelt // t'gewelt // der sinnen,
        Van binnen // s'herten gront,
        Want gene
        Zo reene,
        Allene, gij cont
        
          Doleuren
        
        Doen scheuren,
        Voor treuren // terstont.
      
      
        Al dijn vroetheyt // goddinnelick,
        Zo hemelsch af gemaelt,
        Is een soetheyt // gans minnelick,
        Te vollen nooit verhaelt,
        Mij daalt // en straelt // ter herten
        Veel smert // en // ongeval,
        Achermen!
        Mijn kermen
        Ontfermen // wilt al,
        Of sterven,
        Bederuen,
        Duyst-werven // ik zal.
      
      
        Rancxken aerdich // moet noot // en // recht,
        Doch zoet van aarde // zijt,
        Mij (onweerdich verstooten // knecht)
        Een sauegaerde // zijt:
        Naar strijd // zendt tijd // van vreden,
        Zijt heden // in mijn cruys
        Soet- // mondich,
        Goet- // grondich,
        T'bloet- // hondich // gespuys,
        Laat drijuen,
        Zijn blijuen,
        Catijuen // confuys.
      
      
        Int versuchten // naar troost // en // raad,
        Ligh ik gans kercker // vast,
        Nu mijn duchten // ten boosten // gaat,
        Den brand noch sterker // wast,
        Dien last // deurtast // mij schandich,
        Vyandich // met veel grief:
        O rare,
        Troostbare,
        Int sware // miskies,
        Van dezen,
        Wilt wezen,
        T'genesen // schoon lief.
      
      
        Als idoone // Princess // mij
        Int laetste secoureert:
        O mijn schone // Deesse // vrij,
        Mijn hulpe mainteneert:
        Cesseert // geleerd // verstandich
        Mijn brandich // lijden straf:
        Of! zonder,
        Groot wonder,
        Ik onder // int graf
        Moet dalen:
        En halen,
        T'betalen // hier af.
