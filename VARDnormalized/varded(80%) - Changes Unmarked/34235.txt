
    
      
        GEbeden // seyndt heden // dijn reden // tot God,
        Om swijgen // van ’t hijgen // dat krijgen // verdwenen,
        Moordt, branden // ons’ landen // met schanden // bespot,
        Daar lusten // ons kusten // in rusten // voor-henen;
        Wilt spenen // o Here! ’t vermeerende kwaad,
        Ons zonden // ontbonden // dijn wonden // tot smaedt,
        Maar behoed ons // en doet ons ghenaed’.
      
      
        Wilt sterken // de werken // met merckende reen
        Der staten // die ’t kwaad en // het haten mach scheuren,
        Verlichtse // en stichtse // ja vlechtse in een,
        Dat sy-lien // en wy-lien // ons vrij zien // van treuren:
        De deuren // wilt sluyten // van buit ende roof,
        Bevrijdt // voor den strijdt // en den nijd // het Geloof,
        Maakt voor stoorders // de hoorders doch doof.
      
      
        Wij stromen // in dromen // door ’t schroomen // verblindt,
        Noch soecken // noch boecken // verkloecken de sinnen,
        Ons machten // ons grachten // ons krachten // zijn wind,
        God stuert // en bemuert // ons gebuerten van binnen;
        Gewinnen // ons voeten // t’ontmoeten // der deugd,
        Wij kleven // aan ’t leven // verheven in vreugd,
        Dat altijdt // ons bevrijdt, en verheugt.
      
      
        O hogen // vermogen // mee-dogenden Al!
        Dijn handen // verpanden // de banden // van vreden,
        Worpt neder // ‘t onweder // dat seder // ’t geval
        On-spoedigh // verwoedigh // ons bloedigh vertreden:
        Dijn leden // beschermt // en om-ermt // uwe Bruydt,
        Die ’t smachten // heel nachten // in klachten // weent uit,
        Laat doch broeyen // d’on-bloeyende Spruyt.
      
      
        De wolcken // vertolcken // de volken // den Peys,
        ’t Verhoopen // staat open // om kopen genade;
        ’t Betrouwen // drijft rouw’ en // ’t benouwen op reys;
        ’t Verschillen // der willen // zal stillen // ter stade;
        Te spade // ’t geluk // in den druck // nooit verscheen:
        Bidt krachtig // eendrachtigh // ’t waarachtige Een
        Zal ons geven // te leven // in Vreen.
