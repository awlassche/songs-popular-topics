
    
      
        COR agieuse // violette //
        Reynste fleur // Op aerderijck,
        Amoureuse // schoon Jenette //
        In den geur // aromatijck,
        O soetheyt zonder gelijk //
        Zulk een jeugd //
        Gij zijt //
        Die vermeugt // Altijd //
        Smert en leed
        te scheu- ren //
        Met vreugt //
        Jolijt //
        Verheugt //
        Verblijd //
        Mijn langmoedich treuren.
      
      
        Nobel herte // borght van trooste,
        Acht doch cleen // twistich erreur:
        Doen mij smerte // d'alderbooste,
        Gij alleen // tooght mij faveur,
        Vwen lof, werd met honneur,
        Honderd fout
        Verclaert,
        En als goud
        Bewaard,
        Gants uwen persoone:
        Benout,
        Vervaert,
        Verflout,
        Bezwaard,
        Ik mij t'uwaert thoone.
      
      
        Enig wezen // hier ter weerelt,
        Wel gespraect // en niet te min
        Hoog geprezen // rijk bepeerelt,
        Gans volmaect // in mijnen sin:
        Wtvercoren vant begin,
        Alle mijn
        Verdriet,
        En gepijn
        Is niet,
        Troost gij mij behendich,
        Macht zijn,
        Mij biet,
        T'aenschijn,
        En ziet
        Mijnen staat ellendich.
      
      
        Lieuer steruen // naar sentency,
        (Dit zal elk // houden gewis)
        Dan te deruen // u presenty,
        Zonder welk // geen leven is:
        Welgeboren maget fris,
        Hoe cruel
        Het zij,
        Even wel,
        Zult gij
        Lief! zijn mijn egade:
        Niet fel
        Tooht dy,
        Omhel-
        sen mij
        Laat uwe genade.
      
      
        Int beminnen // ik hert-grondich
        Roupe voort // om hulpe groot,
        Want mijn sinnen // vast, en bondich,
        Zijn accoort // tot in de dood,
        V te laten t'geender noot:
        O dierbaer
        
          Parceel
        
        
          Wederpaer
        
        En deel
        Tot mijn treurich clagen,
        Coemt naar,
        Geheel:
        Dit zwaar
        
          Goreel
        
        Helpt mij lijdsaem dragen.
      
      
        Aldervroomste // schoon Princess,
        Niet onreins // en gaat u bij:
        Mijn voornoemste // forteresse,
        Van vileyns, en dorpers vrij
        Blijft nu voort: hier mede wij
        Scheiden wt,
        Dijn hof,
        Mijn virtuyt
        Hier of,
        Schijnt treurich verdroogen:
        Schoon bruydt,
        Oorloof,
        Mij ruydt
        En grof
        Neemt in u gedoogen.
