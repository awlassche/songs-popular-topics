
    
      
        HEer uwen name // is groot eersame,
        Wijdt uytghebreyt // is uwe fame,
        V Majesteyt // vol heerlijckheyt.
      
      
        Heer u ghenade // duert vroeg en spade,
        Zonder verganck // eeuwig ghestade,
        Even zo lanck // zij u lof danck.
      
      
        Here uwe weghen // zijn al te deghen,
        Altijdt tot goed // zijt gij gheneghen,
        Lof u zijn moet // in overvloed.
      
      
        Heer uwe werken // die doen ons merken
        Dat gij altijd // niet om verstercken,
        Almachtig zijt // ghebenedijt.
      
      
        O Heer u wezen // moet zijn ghepresen,
        V goedheid heeft // ons al ghenesen,
        O Heer gij gheeft // daert al bij leeft.
      
      
        O Heer u woorden // ons met accoorden
        Wijzen hoe dat // wij al behoorden
        Te gaan den pad // na u schoon Stad.
      
      
        V hulp, o Here // hier te allen kere
        Behoeft ons wel // want wij zijn teere,
        Dus maakt u snel // ins ons bevel.
      
      
        Heer, uwen wille // vrij van gheschille,
        Leert ons al doen // van zonden stille
        Als ranken groen // op elk saysoen.
      
      
        O Heer, u Rijke // daar toe ghelijcke
        Ons weerdig maect // dat niemand wijcke
        Verloren naakt // maar neerstigh waeckt.
