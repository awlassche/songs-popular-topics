
    
      
        PAradijs // vergier van lust,
        Die alle twist paiseert,
        Zo propijs // dat nergens ruste,
        Dan bij u en logeert,
        Wat mij smert // miscomt of deert,
        Bij u werd // zulks gecureert,
        Daar mijn hert // om lamenteert.
      
      
        Is mij oyt // geweest indachtich
        Op lieuer t'zijn gesint:
        Liefste nooit // dat ik warachtich,
        Gij strijd, en overwint,
        Naar den treyn // der liefden blint,
        Sijt certain // gij Venus kint,
        Met een rein // herte bemint.
      
      
        Eenich beeld // gulden monstrance,
        Oorspronck aller iolijt:
        Daar gij speelt // vrolick ten dansse,
        Vervreemt discoort, en strijd:
        Schoon Venus // ghebenedijt,
        Ik u dus // ziende altijd,
        Weert een cus // in eeren zijt.
      
      
        Troostbaer lief // ia gratieuste,
        Die mij houdt in ghebiet,
        Al mijn grief // compassieuste,
        Genadelijk aanziet:
        Onder // mij // en is doch niet
        Zonder // dij // dan groot verdriet,
        Wonder // bij // dijn troost geschiet.
      
      
        Eewich zal // ik // onbedwongen,
        Zijn uwen Onderdaen:
        Of t'getal // van duizend tongen
        Mij schimpich wederstaan:
        Gij alleen // Lief, staat mij aan,
        Lieuer geen // secous t'ontfaen,
        Dan u een // voet af te gaan.
      
      
        Redene rijckste // schoon Princess,
        Heuslick treck ik van hier:
        Magnifijckste // in noblesse,
        Schoon bloeyende Vergier,
        Weest doch mijn // eenich plaisir:
        O divijn // maagd goedertier,
        Ik zal zijn // uwen Princier.
