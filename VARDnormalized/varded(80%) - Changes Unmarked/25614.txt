
    
      
        Dancklijck eenmoedich // prijst God den Vaer,
        Want hij zeer goedich // is voorwaar,
        Zijn  gonst om prijzen // gaat hij eenpaer
        Ons nu bewijzen // int openbaer,
        Och vrienden gij //  zingt doch met mij
        Van herten bly // dankt doch God allegaer
      
      
        Int huis des Heren // zullen bereid,
        Wij gaan met eeren // zo hij toeseyt,
        Om vreuchd’ verwinnen // hier boven gaan,
        Ons voeren binnen // de poorten staan,
        Blijft op den ganck // u leven lanck,
        Al zijn wij krank // die hop’ v’steect ons saen
      
      
        Een stad ghebouwen // heerlijck en breyt
        Haar Borghers houwen // die  eenicheyt,
        Men mach bij trouwen //  zeer hoog gesteld
        Haar licht aenschouwen // het lucht int veld
        Och vrienden jent // maakt ook bekent,
        Dat gij nu bent //  borgers elk als een held.
      
      
        Recht doch u wezen // als Christen goed,
        Aan Ziel ghepresen // hert en ghemoet,
        Als de eerste deden // met Ziel en lijf,
        Eenich in vreden // te samen stijf,
        Gods glory schoon // spant daar de Croon
        In s’Hemels Troon // zal blincken haar bedrijf.
      
      
        In deze mueren, eert God den Heer,
        Lieft zonder trueren // elckandren meer,
        Gaat opwaarts beden // voor God aldaar,
        Drijmael t’gheeft reden // int nieuwe Iaer,
        Hem offer bringht // dankt, loeft en zingt,
        Dit vier ontsinckt // de liefde in ons claer
      
      
        Climt op de graden // volheid der deucht,
        Dat gij gheraden // t’gheloove meucht
        Betuigen waerlijck // als volk Gods vroet
        Wt God schijnbaerlijck // ghesproken zoet,
        Met kracht en daad // zo Christus raad,
        Elk huis bestaat // dat men zo bouwen doet
      
      
        Kent, wilt ghevoelen // daar zijn gheset
        De Rechter-stoelen // na rechter Wet
        Oordeel te geven //  in Davids huis.
        Na een elcx leven // zonder abuys
        Crijcht elken slecht // daar loon na recht,
        D’onnutte knecht // zal daar worden confuys
      
      
        Perfeckten vrede // soo’t God behaagt,
        Crijcht niemand mede // dan die nu draagt
        Liefde te voren // int herte hier bij,
        Rein uit vercooren // zonder  party,
        Godlozen ziet // en hebbens niet,
        Want vrede diet // Christus den zijnen vrij
      
      
        Ik wensch u tezamen // vrede van God,
        Liefde met namen // t’nieuwe ghebodt,
        Naar Schrifts verclaren // Ierusalem
        Wilt openbaren // volgt Christi stem,
        In deucht gaat voort // al na zijn woord,
        Niet anders hoort // en zal behaghen hem.
      
      
        En in u toorens // breete ende smal
        Moet zijn te voorens, vred’ over al,
        Daar men behoedich // de wapens hout,
        Bereid en spoedich // zeer menichfout,
        Om de trompet // te blasen net,
        Zijt gij gheset // t’volck Gods is betrout.
      
      
        Treedt op u voeten // t’gheluydt nu staat,
        Ontwaken moeten // zij niet te laat,
        Den vred’ wilt preken //  straft nu bij tijd,
        Zo onbesweken // u Ziel bevryt,
        Gods volk certain // de kudde cleyn
        Bewaard nu rein // want gij haar wachter zijt.
      
      
        En  wassen, groeyen // moet gij Gods stad
        En voerich bloeyen // te wensch u dat,
        Om t’lof te bringhen // voor Gods aenschijn,
        Is nu het singhen // en t’wenschen mijn
        Vreedlijck vergaert //  Godtlijck gheaert,
        Zo Schrift verclaert // met Christen sinnen fijn.
      
      
        Recht op’t vervallen, bindt met der hand,
        Heer, onder allen, des vredens bant,
        Bouwen, vergaren // zouden wij sterk,
        Als trou dienaren // zo aan u Kerck,
        Iae als een man // timmeren dan,
        Gheen vijand can // ons dan beletten t’wercl.
      
      
        Staat om verhogen // Heer bij met kracht,
        Want onvermoghen // is ons macht,
        Gheensins verblijden // en moghen wij,
        Helpt gij hier strijden // aan onze zij,
        In u macht staat // ons kracht en daad,
        Verwinnen laat // ons doch niet moede bly.
      
      
        Zaligheid nacy // gij Christen schoon,
        Zij u, zoekt stacy // in s’Hemels troon,
        Dat gij met namen // zonder beswijck,
        Zo meucht versamen // int Hemelrijck,
        Liefdens vermaen // heeft dit gedaan,
        Neemt het doch aan // van mij zo al gelijk.
