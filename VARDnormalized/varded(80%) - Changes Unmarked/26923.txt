
    
      
        O Here God van machten // groot,
        Voor u zijn mijn ghedachten // bloot,
        Daar en is niet zo erghen // man,
        Die hem voor u verberghen // can,
        Want gij binnen en buiten // ziet,
        Men can voor u opsluyten // niet.
      
      
        T'is voor u als den lichten // dach,
        Gheeft in mij dat wel stichten // mach,
        Enen nieuwen bequamen // gheest,
        Met al die uwen Name // vreest,
        Dat ik mach hier beneden // pleyn,
        Wandelen in der vreden // rein.
      
      
        Zonder den vrede zeer dier // baer,
        En zal men moghen zien hier // naar,
        God in zijn eeuwich wezen // fijn,
        De knecht zal niet ghepresen // zijn,
        Die met onvrede quellen // gaat,
        En zijn medeghesellen // slaat.
      
      
        Met stocking oft met roede // stijf,
        En meen ik op dat goede // lijf,
        Maar met der tonghen en haesten // praet,
        Spreken van zijnen Naasten // kwaad,
        Zo niet der tonghe griefde // rat,
        Maar niet en leert de liefde // dat.
      
      
        Al trockt gij ook een bloedich // mes,
        Tegen u vriend die noodich // es,
        En u goed principalijck // riet
        Zo doet gij noch qualijck // niet,
        Als dat gij met versmaden // maakt,
        Hoe men in cromme paden // raeckt.
      
      
        Die zijnen mond tot laster // went,
        Den vred' hij veel t'onvaster // kent,
        Want naar Syrachs ghewaghen // goed,
        Hij doet van al zijn dagen // boet,
        Die hem tot qualijck spreken // gheeft,
        En selue vol ghebreken // leeft.
      
      
        Here bewaerdt mij in de // pad,
        Naar die reine beminde // stad,
        Mijn tonghe en mond beneuen // dien,
        Dat ik hier naar mach t'leuen // zien,
        En goede dagen mede // voort,
        Och Here mijner bede // hoort.
      
      
        Mijn ghebedt, Heer, verhoren // wilt,
        Dat daar deur uwen toorn // stilt,
        Over alle mijn zonde // boos,
        Door u ghenade, gronde // loos,
        Vergheeft mij die al tezamen // vroet,
        Want ik mij voor u schamen // moet.
      
      
        Laat mij, Here voor u vri // gaan,
        Van zonden, als den Publi // caen,
        Die voor u in den Tempel // bat,
        Gheeft mij naar dat exempel // wat,
        Dies gij naar u verclaren // vast,
        Kwijt maecktet van den zwaren // last.
      
      
        Of ik u zo behaechde // reen,
        Heer ghelijck als Magde // leen,
        Doe zij verkoos dat beste // deel,
        Scholdt gij haar kwijt de rest // heel,
        Der zonden // tot versoeten // dies,
        Zij, met tranen u voeten // wies.
      
      
        Veel biddens, zij met tranen de,
        Ook dat Vrouken van Canane
        Zij bad met goed betrouwen // snel,
        Voor haars dochters benouwen // fel,
        Ootmoedich en van herten // leech,
        Van u zij haar begheerten // creech.
      
      
        Haar dochter deedt gij, Heer, bystandt,
        Zij was ghequelt van den vyandt,
        Deed' haar veel persequeren // an,
        Maar gij sacht dat verneren // van,
        S'moeders ghelooue machtich // sterk,
        Zo toondet gij u crachtich // werk.
      
      
        Zeer veel crachtighe daden // zoet,
        Here, gij deur ghenaden // doet,
        Wie tot u met ghelooue // quaem,
        Melaatse, Creupel, Dooue // Lam,
        Den Blinden sach, den Stommen // sprak,
        Gij hebt van haar genomen // t'pack.
      
      
        Och Here God barmhertich // Vaer,
        Den Man die achtendertich Iaer
        Op't Bedt hadde gheleghen // krank,
        Hebt gij cloeckheyt en deghen // ganck,
        Wt liefden hem goed rond // ghejont,
        En maecktet hem terstont // ghesont.
      
      
        Princelijcke God diuijn // zijt mijn
        Een eeuwich Medecijn // met wijn
        En oly mij inwendich // bint,
        Waar dat ghy't, Heer, ellendich // vint,
        In mij een gheest verslaghen // schept,
        Daarin gij goed behaghen // hebt.
