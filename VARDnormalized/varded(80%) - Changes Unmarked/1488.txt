
    
      
        MEt Brood // met Bier // zeer fijn met Wijn,
        Pastey // Lamprey // met Haes en Conijn,
        Met Hoen // Capoen // Wildbraad // versaet
        Elk is // ghewis // op dit termijn.
      
      
        Ons hert is heel met vreugd // verheucht
        Ons moet // zeer zoet // strekt tot gheneucht
        Wij singhen // en springhen // wij clincken en wincken
        Versterckt o Heer ons jeucht // en deucht.
      
      
        Men ziet hier staan // ghesoon ghebraen
        Die spaert // die Taert // die is belaen
        Mersepeyn // met cleyn // begint het hint
        Die haen // en swaen // can niemand schaen,
      
      
        Wij dancken u veel meer // o Heer
        Dan wij // zeer bly // u schaepkens teer
        met monden // verconden // versterckt en werct
        In ons nu voort // u woord en leer.
