
    
      
        Een dienaar, lichte claer // Godes schaer
        Als ghetrouw, held // stelt,
        Menich jaar // heeft hij haar voorspelt,
        T'oordeel zwaar // hoe't zal zijn ghevelt,
        En vertrokken // met wat wapen.
        Dat die bokken // van de schapen,
        Ras en snel // totter hel // wreet en fel
        In den eewighen gloet // moet,
        T'sal zijn wel // lustich ende zoet,
        Die 't bevel // van den Here doet.
      
      
        Hier na hoort // zijn geboort // sach men voort
        Anno vijfthien // schien,
        Nae't accoort // metten vromen lien,
        Menich oort // heeft hij moeten vlien,
        Men bedreven // d'uytvaert clachtich,
        Als men schreven // tweentachtich,
        Doen hij smertich // had goedthertich // een en dertich
        Iaer dat God verlient // dient,
        Duizend partich // ende die ghetient,
        Mede wartich // met Christo verient.
      
      
        Hij blies den horen // om te spooren // Godes tooren,
        Dien hij over t'lant // sant,
        Wel te voren // dreychde hij Godes hand,
        Snel verlooren // ginckt aan elken cant,
        Want zijn reen // die met beven,
        Van't ghemeen // volk naet leven
        Sou ghevaet // zijn met raad // is versmaet,
        En niet op ghepast // vast
        Godes raad // is meenich een last,
        Die zoet praet // is haar liever gast.
      
      
        Een goede strijd grote spijt // in zijn tijd,
        Heeft hij wel ghestreen // leen
        Claer verblijd // vrymoedech ghetreen,
        Den loop wijt // totten einde reen,
        T'gheloof ghesont // int benouwen
        Tot alder stond // vast behouwen,
        Dat met recht // zij ghesecht // trouwen knecht
        Nu zal zijn die croon // t'loon,
        Niet en vecht // tegen deze troon,
        En voorlecht // t'volck veel woorden schoon.
      
      
        Laat doch af // dat gij caf // ende draf
        Heerlijck voor go weyt // spreyt,
        Een ander straf // u nu toebereid,
        Denckt om t'graf // dat gij niet en schreyt,
        Als die Heer // comen zal bezoeken,
        Werk en leer // na luydt der boecken,
        Dat eersaem // uwen naam // na 't betaem,
        Werd gevonden daar // claer
        Te zijn bequaem // als ghetrou dienaar,
        Met hoogher faem // in Godes bewaer.
      
      
        Ernstich weest // ende leeft // na den Gheest
        Deze spreuke claer // naar,
        Aldermeest // dien hij was te zwaar,
        Ende priest // hem noch openbaer,
        Kwaad kwaad // seytmen int leven,
        Twelck staat // Proverbi twintich schreven,
        Alst vergaen // van die baen // en gedaan
        Is, zo prijstment weer // zeer,
        Wien vermaen // eyscht dan daedt bij leer
        Moet vergaen // ja ook self die Heer.
      
      
        Noemter een // int ghemeen // die met vreen
        In zijn tijd gheleeft // heeft,
        Is er gheen // mij ghelijck hier gheeft,
        Zoekt gheen leen // niet langher en sneeft,
        Moyses leed // dreyghen om te stenen
        Elk Profeet // most dikwijls wenen,
        Waarom want // alle schant // door het land
        Men al den Profeet // weet
        Weg van cant // een yder creet,
        Wie haar hand // boodt creech mee een smeet.
      
      
        Ere groot // strack men boot // na die dood
        Die men dee ontgaen // traen,
        Menich stoot // heeftmens aenghedaen
        Nu isset noot haar weg moetmen gaan,
        Och hoe groen // wie zal does ghesellen,
        Schricken doen // voor die pijn der Hellen,
        Want waar met // niemand bet // weet Gods Wet,
        Elk vermaent met spoet // zoet
        Haar ghebedt // ende werken goed,
        Als trompet // men ook weten moet.
      
      
        Roemt niet meer // van een leer // die ter neer
        Door u zelfs ghestort // wordt,
        Zoekt gheen eer // om dit leven court,
        V begheer // tot het eewich port,
        Dat men u niet // gaat verwijten,
        Dat gij ziet // als d'ypocrijten,
        Ghemaniert // dat gij viert // en versiert,
        Der gherechter lijck // dijck,
        En u tiert // als van deuchden rijk
        Bedroeft gheviert // wordt bevonden slijk.
      
      
        Toont doch niet // dat men u ziet // met verdriet,
        In des Werelts doen // spoen,
        Dat gij u biedt // sterk // met woorden coen,
        Ende vlied // t'werck // om smenschen soon,
        Die wereld tiert // en heeft behaghen,
        Ia zij viert // die Heylighe dagen,
        Prijst haar leer // met begheer // ende eer,
        En die nu noch die daedt // vaet,
        Ende weer // oprecht leven gaat,
        Drucktse neer // ende wordt ghehaet.
      
      
        Bovwen gij // ook als zij // even bly,
        Den want in valscher schijn // blijn
        Spreket vrij // het is die medecijn,
        T'bleeck hoort mij // aan haar kwaad fenijn,
        Israel // die zo ginck leven
        Is rebel // gants verdreven,
        Wt haar land // t'alder schant // haar vijand
        Veel haar bloot // vergoot // root
        Berooft verbrand // met een slach of stoot,
        Dat niemand // oyt sach meer der noot.
      
      
        Sulcken noot // hoewel groot // ende snoot,
        Daar God Israhel // snel
        Mee besloot // grouwelijck en fel,
        Was maar bloot // enen schadu spel,
        T'ware wezen // is gheleghen
        Zo wij lezen // Marcus neghen,
        T'worm niet sterft // men beerft // en bederft,
        Enen eeuwich vuer // suer,
        T'welck doorkerft // gheest ende natuer,
        Niet meer verft // vreest t'eeuwich ghetruer.
      
      
        Och hoe veer // vlied dan eer // als men teer
        Al des werelts staat // laat,
        Aan den Heer // t'ontijt zoekt ghenaet,
        Och hoe zeer // weynich die 't recht vaet,
        Onder die 't // woord Gods dienden,
        Menich niet // die't recht minden,
        Immers gheen // hoort mijn reen // dan alleen
        Timotheus daar // waar
        Paulus gheween // ende nu ook zwaar
        Achtme cleen // nauwelijcx een haar.
      
      
        Enen staat // ende raad // ghelijcke daad,
        Als vant begin ghesweeft // heeft,
        T'selfde maet // men als nu noch gheeft,
        Even kwaad // als voor die mense leeft,
        Zo treffen necht // alle die Schriftueren
        Effen recht // tot dezer ueren,
        Ia all' ellent // en torment // God haar sent
        Die nu zo onbedacht // lacht,
        Die hem ook went // uit alle zijn kracht
        Blijft ongheschent // want God hem wacht.
      
      
        Neemt in danck // dezen sank // u verlanck
        Zij zo veel gij meucht // deucht,
        Poocht u ganck // d'outheyt als die jeucht,
        Om eer lanck // t'erven d'eeuwighe vreugd,
        Crijcht gij iet // hier bedenken
        Aan dit lied // wie't mach schencken
        Iou bericht // gheeft u licht // dit ghedicht
        Noemt hem in dit vars // dwars,
        Al diet sticht // vast staat // klemt als hars
        Int kwaad swicht // ontgaet // Satans pars.
