
    
      
        AL, mal// zijn wij nu te samen
        Go val// heeftet dat wij hier kwamen
        Wij zeggen vrou Bruyt Amen
        Met namen// in deze Feest
        Vrou Lors ons Bruyt// vaet wel tbeduyt
        Is niet bevreesd.
      
      
        Haar Maechdom meest// en zal men nu niet sparen
        Zij heeft geweest// al over langhe jaren
        Veel Dansen op haar snaren
        En paren// met Vrou Lors bekent
        Het goed gaat deur// adieu bonneur
        Zo valt het ent.
      
      
        Zeer excellent// gaat Ioncker Mors nu prijcken
        Na ons Zotten ient// zo gaetmen dapper kijcken
        Veel comter hier ook strijken
        Al gelijcken// dienen zij geschrapt
        Men ziet zonder spot// hier menigen Zot
        Al ongecapt.
      
      
        Wij hebben gehapt// van het gebraen verheven
        Tbier ook getapt// gedroncken boven schreven
        Wilt Vrou Bruyt u begeven
        Zonder beven// te gaan na tbet
        V achterpoort// zo dat behoort
        Staat fraay en net.
      
      
        Prinse besmet// zijn der veel hier ter spacy
        Zij dopen vet// en staan in vrou Lors gracy
        Al lijden zij eylacy
        Temtacy// het en schaet haar niet
        Wie dat gij zijt// als nu ter tijd
        Toch voor u ziet.
