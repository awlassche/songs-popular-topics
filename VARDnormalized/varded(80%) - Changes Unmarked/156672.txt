
    
      
        Ik belijt // het met vlijt // op een tijd
        Mij hebben met reen // been,
        Twee verblijd, om een gave cleen,
        Op dat sy't ghebruyckten ghemeen,
        Enen sank // wast begheeren,
        Ik dorstet lanck // niet consenteeren,
        Hierom want // mijn verstand // alle cant
        Ghelijcke veerdig niet // ziet,
        Nader hand // ik mij noch beriet,
        Ende vant // dit ghetuyghe lied.
      
      
        In de leer // die de Heer // zijn volk teer,
        Heeft tot een ghebedt // set,
        Haar begheer, vervullende net,
        Eyscht hij eer // van elk onbesmet,
        Namelijck dit // zegt te gader,
        Als gij bidt // onze Vader,
        Gij eersaem // uwe faem // ende naam,
        Gheheylicht alle tijd // zijt,
        Onbequaem // hen hier in verblijd
        Zonder schaem, die gantse wereld wijt.
      
      
        Deze gront // werd vercondt // door die mond
        Van des Heren boon // schoon,
        Te aller stond // ons tot enen loon,
        Om ghesont // crijgen t'eeuwich loon,
        Israhel // Gods lof songhen,
        Doen zij snel // eerst ontspronghen,
        Door die zee // alletwee // twelck haar dee
        Farao doen bij gram // kwam,
        Ende mee // haren weg aennam,
        Wijt en bree // gants te grond hij swam.
      
      
        Tot does leer, // tot Gods eer // noch veel meer
        Int Psalm boek vervaet // staat,
        Alle weer // Davidts gantse raad,
        En begheer // alle hier op staat,
        God te prijzen // groot te maken,
        Te bewijzen // in veel zaken,
        Hoe verheucht // watte vreugd // ende deucht,
        Men in dezen oort // spoort,
        Zo hem veucht // hier tegen discoort,
        Die lust der jeucht // dempt met s'heeren woord.
      
      
        Seraphin // Cherubin // ook niet min,
        Haar om zulks te doen // spoen,
        Haren sin // strecket ons ten goen,
        Haar ghewin // is al ons behoen,
        Ellick veylich, roept om seer,
        Heylich heylich // heylich Here,
        Onze God // Sebaoth // u ghebodt,
        V glorieuse raad // gaat,
        Ellick tot // deur en ellick straet,
        Hij is zot // die dit niet en vaet,
      
      
        Een ieder een // groot en cleen // die does reen
        Weg uit zijn ghedacht // slacht,
        Wat die meen // is niet en betracht,
        Komt in vreen // als elk Christen lacht
        Hoe gheschien // moet dit prijzen,
        Het bedien // zal u wijzen
        Met accoort // onghestoort // Godes woord,
        Matheus in het vijft // schrijft,
        Deze oort // ons in Christo lijft,
        Die t'hoort // ende daar bij blijft.
      
      
        Niet toe laat // dat men kwaad // inder daedt
        Van u met waarheid // zeid,
        Altijd staat // haar tot deucht bereid,
        Die u smaet // aandoen ende leit,
        Goed te wenschen // haar te stichten,
        Voor die mensen // laat zo lichten
        Vwen ganck // datse danck // vrij en vranck,
        Zeggen onzen Heer // zeer
        Met verlanck // volghen uwe leer,
        Sulcken clanck // gheschiet God ter eer.
      
      
        Daar toe bly // bidden wij // te allen ty,
        Spreekt Paulus den Heer // zeer,
        Op dat hij // u te recht bekeer,
        Zo dat gij // weerdelijck zijn leer,
        Wandelen zult // al u dagen,
        En vervult // na zijn behaghen,
        Met goedheid // dat gheseyt // en verbreyt
        Zij door u zijn prijs // wijs,
        Zijn Majesteyt // heerlijcke logijs,
        Wel bereid // die in deuchden grijs.
      
      
        Enen raad // schreven staat // zelver gaat
        Iohannes vijfthien // zien,
        Na die maet // moetet recht gheschien,
        T'alligaet // zal ik u bedien,
        s'Vaders eer // wordt bedreven,
        Met begheer // als ons leven,
        Dat elk spoet // vruchten doet // overvloed,
        Te worden Christus knecht // recht,
        Hij behoed // in alle ghevecht,
        Ons ghemoedt // hoe simpel of slecht.
      
      
        Al u werk // zwak of sterk // leeck of clerc
        Al tot Godes eer // keer,
        In elk perck // vordert zijnen leer,
        Dat gheen swerck // v duister verneer,
        Zo gij u leven // na t'vleysch wenschen,
        Gaat begheven // als andere mensen,
        Ezechiel // waerschout wel // dat gij snel,
        Maakt des Heren naam // blaem,
        V ghesel // daar door komt tot schaem,
        Die wereld fel // lastert onbequaem.
      
      
        Eer ik sluit // dat beduyt // hoort mij uit,
        Maeck ik tot u tween // reen,
        Dit gheluyt // vervult u ghebeen,
        Gheen menuyt // of het zij ghemeen,
        Mijn begheer // is t'versieren,
        Van Gods leer // in alle manieren
        Deze raad // uwe staat // mede vaet,
        En die een ander dient // mient,
        Paulus laat // Titum zijnen vriend,
        De maet // weten niet verstient.
      
      
        Valse schijn // gheen termijn // latet ijn,
        V woord of werk raad // daad,
        Al ist u pijn // immers oprecht staat,
        Oftser zijn // oft niet die 't aangaat,
        Denckt hier bij // God ik diene,
        Zo zal hij // ghedult verliene,
        Ghehoorsaem // trouw bequaem // aenghenaem
        V haar die u loont // toont,
        Oftse u naam // lastert ofte hoont,
        Dat al tsaem // immers niet verschoont.
      
      
        Mijnen sank // neemt in danck // elc zo lanck
        Dat hij hem verhuert // duert,
        Vwe ganck // onderdanich stuert,
        Dat gheen stank // Gods naam versuert,
        Vwe Heren // ook d'onaerdich,
        Alder eeren // achtet waerdich
        V verblijd // als gij lijt // in does tijd,
        Om gheen misdaad // kwaad,
        Oftme u smijt // en daar toe versmaet,
        Sulcken strijd // is voor God ghenaet.
      
      
        Even wel // niemand sell // zijn rebel,
        Dien God in den gheest //vreest,
        Maar zeer snel // voldoen aldermeest,
        Haar bevel // self die Schrift deurleeft,
        Mijnen lied // of mijn praten
        Achtet niet // maar wilt vaten
        Petrus leer // Paulus meer // God ter eer,
        Bij dat elk u schrijft // blijft,
        Mijn begheer // dan ghenoech beclijft,
        Looft den Heer // bidt dat hij u stijft.
