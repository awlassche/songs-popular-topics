
    
      
        ALderwaerste Carileen,
        Nymphelijn, die ik van herten meen:
        Toeft mijn graeght// soetebloem// brave Maagd,
        Harders kind// van de Hoeders wonder wel bemint;
        O! geestigh dier// vol playsier// stookt mijn vier
        Nimmermeer// doch zo dat'et mij verteer
        In men hart// dat de smart// mij benart,
        Zwaar en groot// en mij geeft oorsaeck totte dood.
      
      
        'k Wil al mijn vermaeck en jeugd,
        Al mijn lust, mijn vrolijckheyt en vreugd
        Nu aan jou// zoete spruyt, heus en Trouw'
        Stadigh aan// graegh besteden, altijd nu voortaan:
        Zo langh de Go'on// uyt'er Troon// mij zo schoon
        En zo vrij// laten leven// zoet en bly,
        Na mijn sin// zal mijn Min// Ã´ Goddin!
        Van u af// wijken nooit, voorse tijght in 't Graf.
      
      
        Daarom allersoetste Bloem,
        Eerbaer Beeldt, vol alle flicker-roem:
        Reine mond// suyver hert, en gesondt,
        Gaeve Spruyt// schiet u botten van weer-liefde uit;
        O! schiet een strael// datse dwael// datse dael,
        En mijn bloed// deur'er kracht verkeren doet:
        Zo zal zijn// dat ook mijn// droeve pijn,
        Zeer versacht// en tot een beter standt gebracht.
