
    
      
        SErubabel wilt horen,
        Zeer wijs // den prijs // hij heeft,
        Den vroukens uytvercoren
        Haar kracht // met macht // verheeft,
        Oft  den Coninck hoog sweeft,
        Den wijn // zeer fijn // om drinken,
        Die wijven // blijven // blincken,
        Boven al dat er leeft.
      
      
        Al wat die mans bedrijven,
        Met raad // met daad // met kracht,
        Dat is al voor die wijven,
        Als zij // haar bly // toelacht,
        Al haar arbeid versacht,
        Veel goud // elk zoudt // hoog houwe,
        Een wel // ghestel // de vrouwe,
        Is noch hoogher gheacht.
      
      
        Roemt wat oyt ogen zagen,
        Wat mense // wensche // mach,
        Van't aenbegin der dagen,
        Waar deucht // of vreugd // gheschach,
        Van Iubal maakt ghewach,
        Van al // ik zal // noch horen,
        Pryeel // casteel // of tooren,
        Zonder vrou ist gheclach.
      
      
        Al int lusthof des Heren,
        Het welk // noch elk // laudeert,
        Was geheel gheen verseeren,
        Maar rust // met lust // vermeert,
        Noch was niet contenteert
        Adam // maar kwam // in rouwe,
        Tot dat // daar zat // een vrouwe,
        Merckt wat een gave weert.
      
      
        Ras liet Abraham zweren,
        Zijn knecht // met recht // accoort,
        Wel neerstich informeeren,
        Hem sant // na t'lant // aenhoort
        Daar hij was van gheboort,
        Hij most // gheen cost // daar sparen,
        Zijn soon // dits t'loon // vergaren,
        Een vrouwe van dien oordt.
      
      
        Een vrome vrouw gheneest fris,
        T'verdriet // gheschiet // becans,
        Een vrouwe die God vreest is
        Een schone // krone s'mans,
        Een gloriose crans,
        Haar reden // zeden // sinnen,
        Verlicht // en sticht // van binnen,
        Isaacx herte gans.
      
      
        Buiten s'lants Iacob diende,
        Den man // Laban // ghetrou,
        Zeven jaar doen hij miende,
        Rachel // dit spel // aenschou,
        Te hebben t'eenen vrou,
        Elk jaar // een paar // hij wenden,
        Als dagen // plagen // t'enden,
        Namt Iacob zonder rou.
      
      
        Esther en Iudith wrochten,
        Gheen man // zulks can // doen bloot,
        Met haar aenschijn zij brochten,
        Vijand // in schant // ter dood,
        Haar wijsheid schoonheid groot,
        Wij lesent // present // beide,
        Vijand // becant // elk leide
        Zo men zeid t'hooft int schoot.
      
      
        Cloeckheyt der vrouwen merkte
        Elk wel // dit spel // geloofd,
        Debora Barack sterkte
        Iael // zeer fel // toeclooft,
        Den Siffera zijn hoofd,
        Ik prijs // een wijze // vrouwe,
        Seba // kwam dra // in rouwe,
        Abel bleef onberooft.
      
      
        Colstel zijns edele gaven,
        Tot een // zulk reen // u keert,
        Laat haar fonteyn u laven,
        Alleen // en gheen, meer eert,
        Lieffelijck gracieus en weert,
        Vroeg spaet // versaet // u wezen,
        Altijd // verblijd // met dezen,
        Proverbij vijf t'so leert.
      
      
        Akkers, huizen, wij leset,
        Dat vaer // ons daar // versiet,
        Een vrouwe die God vrezet
        Met tot // ons God // om vlied,
        Door haar // smelt zwaar // verdriet,
        Zij gheeft // beleeft // hem cleeren,
        Door zij // komt hij // tot eeren,
        Zonder vrou ist doch niet.
      
      
        Recht heeft die preker schreven,
        Van een // met gheen // ghepaert,
        Wee hem in alle zijn leven,
        Die 't goed // met spoet // vergaert,
        Beter is hij verclaert,
        Een paar // can haar // verwarmen,
        Zijn vriend // verlient // hem d'armen
        Zo hij struikelt vervaert.
      
      
        Ach wee is zwaar om lezen,
        T'welck hij // brengt bij // int perck,
        Wedenaers Weduwen Wezen,
        T'begin // den sin // aenmerck,
        T'is wee t'sy leeck of clerk,
        Een wordt // wel court // vergaende,
        Twee zouden // t'houden // staande,
        Een drievout snoer is sterk.
      
      
        Hoort hier na Petrus raden
        Een man // wel can // doort wijf,
        Comen tot Gods ghenaden,
        Die t'woort // verstoort // tot kijf,
        Door der vrouwen bedrijf,
        T'welck niet // gheschiet // met vlechten,
        Noch goud // veel fout // t'aenhechten,
        Om haar uyterlijck lijf.
      
      
        Elk pooch dits mijn vermanich,
        Met vreugd // na deucht // om meest
        Den mannen onderdanich,
        Een sacht // stilach // te gheest,
        Dits hoog voor God geweest,
        Dus pleghen // deghen // vrouwen,
        Eertijds // den prijs // te houwen,
        Als men van Sara leest.
      
      
        Laat haar nu een exempel,
        O wijven // blijven // dijn,
        Al die Gods heylighe tempel,
        Met stenen // meenen // fijn,
        Vercieren als robijn,
        Elk stel // haar wel // te leven,
        V daedt // gheen schaet // laat geven,
        Om haar dochters te zijn.
      
      
        En ghelijck hem verblijdet,
        Een man // al van // zijn bruyt,
        Esaias belijdet,
        Zo zeer // die Heer besluyt, Sach.2,8
        Ons in liefden beduyt,
        Als t'sin // des min // naers blaket,
        T'aenschou // der vrou // vermaket
        Zo die Heer ons viertuyt.
      
      
        Niet alleen voor den Vader
        Door deuchd // does vreuch // de rijst,
        Maar d'enghelen alle gader,
        Van een // onreen // die grijst,
        Want kwaad en boet bewijst,
        Lucas // schrijft ras // van dezen,
        Vijftien // wilt zien // en lezen,
        Tot hem keren hij prijst.
      
      
        Daar toe kwam Christus goedich
        Verbond // ons won // den zeer,
        Als wij onreyn en bloedich,
        Verslaghen // laghen // neer,
        Hij wies, hij salfd ons weer,
        Hij cleed // ons deed // dus krone,
        Zijn trouw // als vrou // we schone,
        Schanck ons met groter eer.
      
      
        En zouden wij niet horen
        Den stem // van hem // zo zoet,
        Komt dochter neigt u oren,
        Verlaat // u staat // met spoet,
        V volk uws vaders goed
        Doet an // ghespan // en dagen,
        Zo fel // zeer wel, behaghen
        Den Coninck u ootmoet.
      
      
        Roert doch die Schrift van wijven,
        Daar meest // Gods gheest // gaat raen
        Der testamenten drijven,
        Hoe 't oud // noch soud // vergaen,
        Maar t'nieu soud eewich staan,
        Sara // sprak dra // drijft schone
        Weg gheeren // dienst deeren // haar soon,
        Die en zal gheen erf ontfaen.
      
      
        Wijven zijn d'instrumenten
        Geweest // self leest // Gods Wet,
        Daar hij zijn Testamenten
        Voorwaar // zeer claer // mee set,
        D'eerste was Lea met,
        Zes soon // heel schoon // beschoncken,
        Rachel // zeer wel // kwam proncken
        Met Ioseph dien hielp bet.
      
      
        Tot verlossing werd zonden
        Den soon // van schoon // Rachel,
        De Zone Gods bevonden
        Een hoofd // dankt looft // hem wel,
        Van 'tgheestelijck Israhel,
        Zijn lust // vrij blust // in deuchden,
        Zijn wil // houdt stil // in vreuchden
        En leeft na zijn bevel.
      
      
        Het zij mannen of wijven,
        Grieck Iued // voorhued // besneen,
        Die bij zijn woord vast blijven,
        Bequaem // zijn tsa // men een,
        Zijn heylighe lichaams leen,
        Zijn lief // zijn brief // zijn tempel,
        Alleen // een reen // exempel
        Aller mensen ghemeen.
      
      
        Merckt ook Aswerus dingen,
        Datsy // liet hij // ghelijck,
        Door zeven camerlinghen,
        Ontbien // om zien // publijck,
        Hoe schoon en suyverlijck,
        Den Heer // zijn eer // te romen,
        Maar stout // zij wout // niet comen
        Is verstooten van haar rijk.
      
      
        Esther t'was een Iodinne,
        In 't goed // ootmoet // bequaem,
        Werd weder Koninginne,
        Ontfinck // alle dinck // daad, naam,
        Dat Vasthi liet met blaem,
        Zij was // dat pas // van staten,
        Niet meer // maar eer // verlaten,
        Maar zij was aenghenaem.
      
      
        Daar staat Remeynen ellef,
        V houdt // niet stout // maar vreest
        Want die natuerlijcke sellef,
        Hoe claer // elk waar // die meest,
        Zijn niet verschoont geweest,
        Veel schilt // wij wilt // doch waren,
        Ons condt // goed rondt // niet sparen
        Dan door een ander gheest.
      
      
        Een ander gheest te deghen
        Wordt claer // Gods schaer // gheleert,
        Van Christo Luca neghen,
        Gheen saeck // met wraeck // verweert,
        s'Gheests vruchten zijn gheeert //
        Veel deucht // als vreugd // en vrede,
        Lanckmoedich // goedich // mede,
        Zulk een Christus begheert.
      
      
        Princess edele vrouwe,
        't Lams bruyt // komt uit // verciert
        V bruydegom ghetrouwe,
        Ghelijck // Cantijck // maniert,
        Suyver ongheschoffiert
        Zijn raad // u laat // ghenoeghen,
        Zijn woord // accoort // wilt voeghen,
        En zijnen Sabboth vyert.
