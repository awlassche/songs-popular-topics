
    
      
        V Vrij verblijd// gij Gods Gemeente,
        Met Christo zijt// gij in vereente,
        Hebt goeden moet// door zijn woord zoet,
        Wie u aan doet// smaet oft vercleente.
      
      
        Het is ghewis// die oude Drake
        Gheneghen is// tot strijdt en wrake,
        Op u verstoort// met fel discoort,
        Maar ziet end' hoort// om wat oorzake.
      
      
        Joannes, in// zijn openbaren,
        Zal u den sin// hier van verclaren,
        T'is om t'bevel// end' Gods instel,
        Zij willende wel// men lietet varen.
      
      
        Oft const gij daar// huychelen mede,
        Gij had met haar// ter weerelt vrede,
        Gheacht, gh'eert// vrij onverfeert,
        Waar gij, verkeert// in land oft stede.
      
      
        Hier tegen dient// bedocht verstandich,
        Dat s'weerelts vriend// God is vyandich,
        Ende zo wie// God is partie,
        Och lacy die// moet vergaen schandich.
      
      
        T'is beter God// den Heer aencleven,
        Want zijn ghebodt// is t'eeuwich leven,
        Hier vrij op acht// zeer wijs bedacht,
        T'volherderns kracht// zal u God geven.
      
      
        Zeer listich can// Satan// bestrijden,
        T'artijckel van// den Ban// end' t'Mijden,
        Op dat// die stad// waar al een gat,
        Oft den muur plat// aan allen zijden.
      
      
        Als op den berg// soect door veel dingen
        Zijn secten ergh// hij noch te minghen,
        Met t'Rijck Christi// aanbidt zeid hij,
        Ik zal u vrij// alles ghehinghen.
      
      
        Veel onghesont// bouwen op dezen
        Ydelen gront// Gods volk ghepresen
        Soud' hier in't dal// breedt over al,
        Onder t'ghetal// der secten wezen.
      
      
        Maar Belial// hoe sy't gloseren.
        Men niet en zal// gheaccorderen,
        Met Christo fijn// in gheen termijn,
        Zij niet en zijn// te compareren.
      
      
        Princess rein// gij s'Lams huisvrouwe,
        V Capiteyn// blijft vrij ghetrouwe,
        Dat gij in deucht// verwerven meucht
        D'eeuwighe vreugd// voor corten rouwe.
