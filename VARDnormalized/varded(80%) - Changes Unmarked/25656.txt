
    
      
        Int Hemelrijc // daar Christus die zal wijzen
        Die hem gelijc // hebben gediend voorwaar,
        Met daad en blijck // in cleeden, drinken, spijzen
        Zonder beswijck // in allen nooden zwaar,
        Nu hier en daar // aan die cleyn Christen schaer
        Is vreugd, zo daar gheschreven staat,
        Die hij den zijnen geven gaat,
        Hoort allegaer // van dezer vreuchden claer.
      
      
        Af nu gheleyt // den dood, en van het sterven
        Verzekerdheid // nu van den tweeden dood
        Verlost, bevrijt // het leven te beerven
        Eeuwich voorzeid // van den zwaren aenstoot
        Wt aller noot // verlost uit liefden bloot,
        Want die dood is verslonden snoo,
        Als zij dit wel deurgronden zo,
        Verheucht devoot // zo wordt haar blijschap groot.
      
      
        Comen dan voort // uit alle p’rijckels nooden
        S’duyvels verstoort // en van der hellen pijn,
        Om met accoort // God te loven der gooden,
        Die naar zijn woord // haar heeft v’lost devijn
        Dit sl den schijn // van haarder vreuchden zijn,
        Verlost uit s’vyandts tanden fel
        Door Christi sterke handen snel,
        Does medicijn // verhoogd haar t’herte fijn.
      
      
        Ook zullen zij // vrolijck zijn en v’hoogen,
        Van herten bly // om datse binnen gaan
        Ter plaatsen vrij // daar zij g’lucksalich mogen
        Eeuwelijck bij // die uytvercoren staan,
        God wel gedaan // eeuwelijck schouwen aan,
        Och wat vreuchde komt daar ontrent
        Boven int Hemels claer convent
        Naar Schrifts vermaen // die zij zullen ontfaen.
      
      
        Blijdlic v’jeucht // zal haar hert zijn ontsteken
        Heel wel v’neucht // om datse Christum schoon
        Vol aller deucht // zullen zien onbesweken,
        T’vermeerdert haar vreugd // boven in s’Hemels troon,
        Die haar ydoon // op set des levens Croon,
        Palmtacxkens vyerich groeyende
        In haar hand tierich bloeyende,
        Dit gheeft tot loon // Christus Iesus Gods Soon.
      
      
        Treuren vergaet // die blijschap zal v’meeren
        Als zij t’ghelaet // der Enghelen ghetal
        Hebben te baet // bij den Here der heren
        Vroeg ende laat // tot gheselschap voor al:
        Vrienden hoe zal // daar vreuchd’ zijn met geschal,
        Als zij bij alle vromen hoort,
        En heylighen daar comen voort
        Van’t jammerdal // bevrijdt van ongheval.
      
      
        Op datse niet // naakt en bloot en v’toonen
        God haar aanziet // het tweevoudighe cleyt
        Des lichaams ziet // eerlijc bij hem te wonen
        Daar over giet // hij nu die zaligheid,
        Maakt ooc bereid // t’verciersel uytgebreyt
        Der zielen // d’welck is hovelijck,
        Gherechticheyt zeer lovelijck,
        All’ dit voorzeid // bringt blijschappe planteyt.
      
      
        Nooit smaec zo zoet // noch lieflijcker aenschouwen
        Dan ons ghemoet // met onzen God certain
        Vereenicht vroet // zal zijn met vast betrouwen
        God met ons goed // die heylighen gemeen,
        Alle groot en cleen // worden te samen een
        ‘Met ons, en wij met haar, en wel
        Zal men daar op het snaren-spel
        Des herten rein // spelen in s’Hemels pleyn.
      
      
        Int hemels land // zeer vast met God v’bonden
        Der liefden bant // ons zo met hem verbint,
        Aan elken cant // zal does liefd’ te allen stonden
        Zeer abondant // wassen als zij begint,
        Seer bemint // staat dit niet in den wind,
        Om gaan tot deze rusten in
        Keert daar toe uwen lust en sin,
        Wordt zo gesint // t’is t’beste dat men vint.
      
      
        Sluytlic hier me // vrienden wilt doch gedincken
        In s’hemels ste // te comen, op den pad,
        Maect u doch ree // dinct hoemen daar zal blincken
        Witter dan snee // ja als der Zonnen rat,
        Dinct vrienden wat // groter vreuchden is dat
        Wie machse doch uytspreken hier,
        T’verstandt moet hem ontbreken schier,
        Soect dezen schat // boven in s’hemels stad.
