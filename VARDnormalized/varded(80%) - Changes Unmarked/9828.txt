
    
      
        SYon verblijden // wilt in u lijden
        'tHert om bevrijden // te Gode wendt,
        Wendt u tot strijden // wilt slapen mijden
        'tZijn kwade tijden // tis bij den endt,
        'tSerpent // ontrent // u te allen zijden,
        Zijn oud benijden // is u bekent,
        Blijft den Here ontrent,
        Zo blijft gij ongheschendt.
      
      
        Waect met verstanden // versterckt u handen
        Laat liefde branden // al meer en meer,
        Vermijdt de banden // en strikken van den
        Sathan, u schanden // begheert hij zeer,
        Verwacht den Heer // in 'sWerelts landen,
        T'en is niemanden // bekent, wanneer
        Dat hij wederkeer
        Zal doen, dus volgt zijn leer.
      
      
        Verblijd met beven // stad Gods becleven
        Op den verheven // berg schoon ghebout,
        Christus u leven // heeft hij ghegheven,
        Dat gij zijt bleven // der aarden sout,
        En weest niet stout // wacht u voor sneven,
        V komt beneven // strijdt menighfout,
        Maar 't veld gij behoudt,
        Beproefd int vier als goud.
      
      
        Zijt gij ghebleken // eens Apoteken
        Reukwerk ontsteken // al offer zoet,
        Zoet int uytbreken // stank vlied geweken
        Die u verspreken // gij stom zijn doet,
        Een boom bij vloed // der waterbeken
        Zijnde gheleken // vruchtbarigh vroedt,
        V bladers tot spoedt,
        Zijn den Heidenen goed.
      
      
        Die woest van zeden // zijn onbesneden,
        Aan d'aertsche leden // 'sherten voorhuid,
        Die sticht gij heden // met rechte treden,
        Te doen na reden // en Schrifts besluyt,
        Zoet reuckigh kruyd // hofken in vreden,
        Stort rein ghebeden // met schalen uit,
        En met Harp gheluydt,
        Zingt lof het Lam als Bruydt.
      
      
        Met 'sherten snaren // dankt God, die sparen
        V voor beswaren // wil, en voor kwaad,
        En openbaren // door zijn dienaren,
        Den reuk, de maren // van synen raad,
        Met woord en daad // in alle scharen,
        Dus om bewaren // dit edel zaad,
        Dat niet en vergaet,
        Altijdts gheneghen staat.
      
      
        Prinse van machten // met u heyrkrachten
        Wij u verwachten // en anders gheen,
        Wee de gheslachten // die u verachten,
        Haar naken klachten // en groot gheween,
        Gheeft ons ghemeen // goede ghedachten,
        Om na te trachten // u woord alleen,
        Ghebout op den steen,
        Vest ons int noodigh een.
