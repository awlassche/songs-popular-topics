
    
      
        Ik zal voorwaar // den Hemelschen vaer,
        Int openbaer // gaan prijzen,
        Zijn lof zeer claer // zal voor en naar,
        Bij zijn dienaar // noch rijsen,
        Die siele mijn // zal met blijde schijn,
        Hem verheughe dijn // o wijze,
        Dat het hore fijn // op dit termijn,
        Die ellendiche dijn // en verjolijse.
      
      
        O Goddelijcke graen // met malcander beraen,
        Met druck belaen // verschoven,
        Laat ons gheraen // met Davids vermaen
        Des Heren naam // gaan loven,
        Doen ik den Heer // noch sochte van veer,
        Hij antwoorde mij weer // van boven,
        En verloste mij teer // door zijnen leer,
        Van alle mijn zeer // mij beroofde.
      
      
        Die hem aanziet // en tot hem vlied,
        Die en wordt niet // tot schanden,
        Want God die biedt // die ellendige met vliet
        Die tot hem riep // zijn handen,
        En helptse voort // al worden ze verstoort,
        Wt alle noot // en banden,
        Zijn engel hoort // beleghert met accoort
        Der gheloovigher oort // spijt vijanden.
      
      
        Ziet ende smaect //watten vriendelijcken saec
        Dat hem ghenaeckt // na dezen,
        Die na God haeckt // en op hem waeckt,
        Niet en versaeckt // maar vrezen,
        Vreest God altijd // en hem belijt,
        Gij die nu zijt // verresen,
        Want die met vlijt // hem vreest en belijdt,
        In dezen tijd // zal zijn ghepresen.
      
      
        Werpt u begheer // alleenich op den Heer,
        Voort kwaad ghebeer, // u wilt hoeden,
        Die hongherde zeer // na zijn Goddelijcke eer,
        Vervulde hij weer // met goeden,
        Maar die hem versmaet // nu zijn versaedt
        Rijk hoog van staat // van moede,
        God weer verlaat // nootdruftich gaat,
        En hem ghenaeckt veel teghenspoede.
      
      
        Hierom verlaat // dit vuyle ghewaet
        En sulcken staat // mispresen,
        Gods vrede aenvaet // die s'menschen raad,
        Te boven gaat ghepresen,
        En gheeft gheen wijck // ist waerachtelijck,
        Oprecht cuysch lieffelijck // int wezen,
        Ist wel luydelijck, ist deuchdelijck,
        Ist lof eerlijck // gheresen,
      
      
        Komt herwaarts hoort // Gods kinderen zijn woord,
        Na waarheid spoort // te deghen,
        Wie is er voort // die na 't behoort
        Met goed accoort // te leven,
        En wie gherust // tot goede dagen heeft lust,
        Iou tonghe uytblust // voor sneven,
        Iou lippen sust // van bedrieghen onrust,
        En boze lust // te streven.
      
      
        Vant boos wilt vlien // doet goed nadien,
        Na vrede allien // wilt jaghen,
        Want God ten tien // wil haar ansien,
        Die hem ghemien // behaghen,
        Ook haar ghebeen // en droevich gheween,
        Wilt de Here alleen // draghen,
        Die haar afscheen // en boosheid deen
        Wil dit Here van een // verjaghen.
      
      
        Die Here voldoet // t'ghebedt zeer zoet,
        Eens verslaghen ghemoet // vol smerten,
        En hij is goed // die tot hem spoet,
        Ghebroken vroet // van herten,
        Die gherechticheydt // voor die mensen verbreydt,
        Van t'quaet hem scheyt // met moeden,
        Hoewel hij schreyt // hem zijn bereid
        En toegheseyt // veel goeden.
      
      
        Maar die nu lacht // en wordt gheacht,
        Van s'menschen gheslacht // ghepresen,
        En toont zijn macht // over t'gheslacht
        Die uit al haar kracht // God vrezen,
        Die wil God weer // werpen ter neer,
        Zal worden zeer // begresen,
        Door Godes leer // die nimmermeer
        Zal nemen keer // ist bewezen.
      
      
        Hierom gij al // der Christenen ghetal,
        Treedt den weg smal // na t'leven,
        Al rijster bal // druck ongheval,
        Hier in dit dal // der sneven,
        Voor t'cleyne verdriedt // d'welcken hier schiedt,
        En wilt doch niet // zo vrezen,
        Maar hier op ziet // wat God onthiet,
        En ons aenbiet // te geven.
      
      
        Hij roept komt bij // die beladen zij
        Ik zal u maken vrij // van zonden,
        Scheyt uit het ghecry // wilt wezen bly,
        Mijn volk zult gij // zijn bevonden,
        Mijn jok is zoet // dat op u doet
        Mijn last is goed // den gezonden,
        Maar t'is al roet // en teghenspoet,
        Die zijn verwoet // mijn gronden.
      
      
        O Princelijcke God // zijt gij ons deel en lodt,
        Bidden wij in slot // ons sanghen,
        Tegen s'werelts spot // door u ghebodt,
        Sterckt uwen rodt // haar ganghen,
        Vwen gheest zendt // dat wij present,
        T'gheloove jent // aenhanghen,
        En s'Hemels tent // na dezen endt,
        Zeer excellent // ontvangen.
