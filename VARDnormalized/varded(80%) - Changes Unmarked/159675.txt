
    
      
        Laat ons verblijden // met vrezen en beven
        Bozen schijn mijden // want ons is gegheven
        In onze lijden // int begin der tijden,
        Macht om te leven.
      
      
        In zond geboren // als de canenijten,
        En gants verlooren // ginck met ons versmijten
        Al soud wij smoren // niemand can ons horen
        Priester Levijten.
      
      
        Stramen en wonden // waren al ons leden,
        Dood in der zonden // t'hert was onbesneden,
        Onreine wonden // onghesouten gronden,
        Godloose zeden.
      
      
        Bloot ende naken // lagen wij met schanden,
        In alle zaken // was niet voorhanden,
        Maar God liet maecken // rokken wie zal 't laecken,
        Van ons vijanden.
      
      
        En ginck ons proncken // met spansels oorringhen,
        Hij heeft ons schoncken // sijden costel dingen,
        Van goud wij bloncken // int riet gelijc voncken
        Cierlijck wij ginghen.
      
      
        Tot alle nacy // is does faem gheresen,
        Van deze gracy // worden wij ghepresen,
        Maar wij och lacy // hoe van estimacy,
        Wordt hier bewezen.
      
      
        In die Profeten // leeftment overvloedich,
        Hoe't wordt verweten // Israhel stoutmoedich
        Dat zij vergheten // hoe zij had gheseten,
        Onreyn en bloedich.
      
      
        Alle die dieren // cranen ende swalven,
        Halt zijn manieren zijner comste halven,
        Maar te regieren // mijn volk recht te stieren,
        Helpter gheen zalven.
      
      
        Kermen en clagen // hoe God heeft verheven
        Na zijn behaghen // Israhel van leven,
        Maar zij ontsagen // weldaet nochte plagen,
        Tegen te streven.
      
      
        Ossen en muylen // kennen hare Heren,
        Maar ik moet huylen // mijn volk wil niet leren,
        Het brist als guylen // t'graeft malcander cuylen
        Om te verheeren.
      
      
        Priesters en Heren // die't volc zouden straffen
        Zijn stom in't leren // honden die niet blaffen
        T'goet zij verkeren t'boose zij eeren,
        Terwe moet kaf zijn.
      
      
        Soud ik niet wreken // zulke boze daden,
        Ik ginck uytreecken // alle mijn ghenaden,
        Met loose treecken // met hoveerdich spreken
        Sy't al versmaden.
      
      
        Daarom met banden // wil ickse versenden,
        Tot haarder schanden // met groter ellenden,
        In vreemde landen, onder haar vijanden,
        Aenschout net enden.
      
      
        Och wat al plagen // hongher pestilency,
        Mosten zij draghen // daar toe vyolency,
        Zo vele dagen // dat God mocht behagen,
        Haar penitency.
      
      
        Komt weer uit Babel // nu alle ghelijcke,
        Acht niet als fabel // Gods woord en zijn rijke
        Weert den diabel // weest niet variabel
        Maar autentijcke.
      
      
        Hier condy sporen // tweederleye leven,
        Dat was verlooren // ghinck God eerst verheven,
        Daar na met toren // ginck hys al verstooren,
        Die kwaad bedreven.
      
      
        Tot alle beiden // lieven ende vrezen,
        Laat u toch leiden // door singhen en lezen,
        Wiltse recht scheiden // u ook recht bereiden
        Tot 'sHemels wezen.
      
      
        Eermen u dringen // wilt van liefde loopen
        Ter bruyloft ringen // wilt gheen ossen coopen
        Noch onschult bringen // van tijtlijcke dingen
        T'eewich wilt hopen.
      
      
        Recht wordt ghepresen // lieven ende minnen,
        Laat zijn geresen // haar vruchten van binnen,
        Zij zal ghenesen // al uwe vrezen,
        Gij zult verwinnen.
      
      
        Wel hen die't vaten // in een go conscienty,
        Tijtelijck verlaten // doende penitenty,
        Haar leven haten // t'sal haar vele baten,
        In Gods sententy.
      
      
        Enen Eleaser // zeven zonen tellen,
        Enen Beltsaser // met zijn drie ghesellen,
        Nebucadnaser // Antiochus dwaser,
        Woudense vellen.
      
      
        Dood nochte Helle // can die liefde crencken,
        Stromen rebelle // canse niet verdrencken,
        Al woudy tell // al u goed daar stelle,
        Daar voor toe schencken.
      
      
        Want ellick strijder // worstelt in zijn percken
        Daar zal men yder // loonen na zijn werken,
        Kwade met lijden, goede met verblijden,
        Och wilt toch merken.
      
      
        Waarom wilt horen // des Apostels raden
        Laat niet verlooren // den tijd der ghenaden,
        Of Godes tooren // na t'gheclanckx labooren,
        Soud u veel schaden.
      
      
        Eer deze dingen // komt die toecomen moete
        Wilt doch gheringhen // doen oprechte boete,
        Zo suldy singhen // en van vreuchde springen,
        Dit is mijn groete.
