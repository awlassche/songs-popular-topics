
    
      
        IN Esaie // prophecy
        Int tweede staat gheschreven
        Hoe dat het zij // ten laatsten ty
        Als dan zal hoog zijn verheven,
        Den berg ghewis// op welker is
        Mijn groot ghedruys // des Heren huis,
        Als dat er veel hopen // hene loopen,
        Met vlijt // verblijd
        Elk ander dus toe  coopen.
      
      
        Al afghelaen // komt laat ons gaan,
        Tot dezen berg ghepresen,
        Die vast zal staan // nimmermeer vergaen,
        Maar eeuwich t'hoochste zal wezen
        In welk die God // van Iacobs rodt,
        Nu hout present // zijn schone tent,
        Dat hij ons leer te deghen // zijne weghen,
        Zijn paen // te gaan,
        En in gheen vreemde steghen.
      
      
        Niemand aenhoort // noch gheenen oort
        Dan Syon uytghelesen,
        Heeft Godes woord // met goed accoort
        Ierusalem ghepresen,
        Voert Godes wet // die onbesmet,
        Oordeelen zal // die Heydens al,
        In oprechtigher claerheyt // na der waarheid,
        Zijn recht // bevecht,
        Die volken zonder swaerheyt.
      
      
        In sulcken ganck // zal elk te danck
        Zijn swaert tot ploechysers breken,
        Die spietsen lanck // tot sickelen blank,
        En dencken niet meer te wreken
        Van deze gheen // ja ook met een
        Die daar begheert // eenighe zweert
        Tegen iemand te luycken // en ghebruycken,
        Gheen striedt // gheschiedt
        Van haar noch zulke stuycken.
      
      
        Alzo does reet // van den Profeet
        Luydt gatet in onze tijden,
        Want veel is leed // haar overtreet,
        Waar over haar nu verblijden
        Hier niet alleen // Christus ghemeen,
        Maar ook zeer claer // d'enghelen schaer,
        Want zo staat hier beneven // claer geschreven
        Van een // onreen
        Die hem tot boet wil geven.
      
      
        Komt doch eens naar // aenschout die schaer
        Die nu na t'beste haecken,
        Die voormaels waar // s'werelts dienaar
        Betracht nu andere zaken,
        Eertijds was zij // eyghen nu vrij,
        Gheneycht tot kwaad // nu tot weldaedt,
        Gierich oncuys zij zaten // boven  maten,
        Met veel // zulk speel,
        T'welck zij nu al verlaten.
      
      
        Ook ziet hier meer // hoewel noch teer
        Die mee beginnen te comen,
        Al haar gheweer // legghen ter neer,
        Aanzien die mensen als bomen,
        Wijn zonder alst // melk onghevalst,
        Begheeren zij // als kinderen ny,
        Op dat al jonge ranken // door die drancken
        Aemacht // tot kracht,
        Zij comen en God dancken.
      
      
        Peys en free // is in does stee,
        Met God voor zijn ghenade,
        Dus wendt u tree // gij alle nu mee,
        Die vreest der zielen schade,
        Nu isset vrij // komt alle mee bij,
        Het is nu claer // t'recht gulden jaar
        na der Basuynen clangen // wilt verlangen,
        Does dach // vermach
        lElck een zijn erf te langhen.
      
      
        Zal nu does teyt // der zaligheid
        Does aenghename dagen,
        Met goed bescheyt // worden verbreyt,
        En dat na Gods behaghen,
        Sulcx niet allien // hoort te gheschien,
        Met groot begheer // der reyner leer,
        Maar wij moeten niet beven // ooc ons leven
        Daar naar // gants claer
        Met neerstigheid begheven.
      
      
        Op dat wij geen // der swacker leen,
        Tot erghernisse strecken,
        Want vele ween // staan over deen,
        Die zijn des lichaams vlekken,
        Haar straf met spoet // haars naasten bloed,
        Och t'is zeer zwaar // komt al op haar,
        Als vruchteloose bomen // en onvromen,
        Met vliet // toesiet
        Om sulckes te ontcomen.
      
      
        En wilt doch niet // den blinden iet,
        Tot enen aenstoot zetten,
        V leven biet // zonder verdriet,
        Dat elk daar op mach letten,
        Duldich in druyck // noot angst misbruyck,
        In slaghen daar // ghevanckenis zwaar,
        In oproer arbeid vaecken // ook in waecken,
        Het zij // of gij
        Om u gherucht ook raecken.
      
      
        Nimmermeer laat // aan u iet kwaad,
        Met waarheid zijn te wreecken,
        Corinthen staat // ik houde maet,
        Op dat ik andere te preecken,
        Niet self en ben // verwerpelijck slem,
        Dus laat ook wij // waecken bety,
        Dat wij zonder beswaren // recht verclaren
        Ons baen // wilt gaan,
        Want wij zijn Gods dienaren.
      
      
        Gheen oorsaeck gheeft // der onbeleeft
        In deze tijd der ghenaden,
        Dat gij niet sneeft // en hij niet heeft
        Waar mede hij u mach schaden,
        Op dat ons pont // niet ga te gront,
        Maar onze licht // com int ghesicht,
        Op dat ghelijck wij lezen // ook bij dezen,
        Gofs naam, eersaem,
        In ons hoog werd ghepresen.
      
      
        Alsulcken leer // eyscht God die Heer,
        Die beyt met woorden en werken,
        Zoekt zijne eer // cloeck is te weer,
        Zijn teere spruytkens te sterken,
        Deze met deucht // verwekt die jeucht,
        En maect verblijd // alle die belijt,
        Christum en zijnen vader // drucx ontlader,
        Doet goed // met spoet,
        Gij Christen alle gader.
      
      
        Voor zulke vreugd // voor zulke deucht,
        Voor zulke grote ghenade,
        Die ons verheucht // al dat gij meucht,
        Lovet God vroeg en spade,
        Wie zal ghenoech // in eenigher voech,
        Lof prijs en eer // doen onzen Heer,
        Van deze schone zaken // vant vermaken
        Doort zien // van dien,
        Die totter kennisse raken.
      
      
        Tot lof en danck // der lippen gheclanck,
        Laat ons gheheelijck begheven,
        Vrij ende vranck // met vrolijck ghesanck,
        Altijd zo langhe wij leven,
        Ook hert en sin // van dit ghewin,
        Want hij is goed // die't wassen doet,
        Bidden dat hij ons soar // en beware,
        Na dit ghecrijt // met hem subijt,
        Ons hem ghelijck verclare.
