
    
      
        V wel bekwame // Bruydt,
        Door uwen Name // uit
        Helpt rechten wonder // dingen,
        In u macht, sterken // Heldt,
        Men 'svyandts werken // veld,
        En al kan te onder // bringhen.
      
      
        In uwe grote // macht,
        Eer ik de bloote // slacht,
        Helpt mij toerusten // stadigh.
        Want tis van noode // wel,
        Op dat ik dode // snel
        Altijdts de lusten // schadigh.
      
      
        T'gheloove t'eenen // schilt
        Mij altijdts leenen // wilt,
        Want meer dan stalich // dezen
        Mij dient in elken // tijd,
        En ook in welken // strijdt,
        De hop' om saligh // wezen,
      
      
        Dien Hellem vryde // mij
        Ook aan mijn zijde // zij
        'sGheests zweert, de waarheid // mede,
        Tis enen goeden // Kreeft,
        Die rein ghemoeden // gheeft,
        In suyver klaarheid // vrede.
      
      
        In liefde vyerigh // wis
        Die goedertierigh  // is,
        Wilt mij rechtsinnigh // gorden,
        Mach zij mij blyven // bij,
        Ik zal beklyven // vrij,
        En overwinningh // worden.
