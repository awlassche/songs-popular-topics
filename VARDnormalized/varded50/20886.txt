
    
      
        GEkomen // vernomen
        Is heden in het stal
        Den Hoeder // den Voeder,
        Den Schepper van all':
        De Knapen // der schapen,
        Verlaten hun Vee;
        En zoeken // in doeken
        Blijmoedig Hem dree.
      
      
        Haar kelen // en Velen
        Die worden gesteld;
        Zij zingen // en springen
        Van vreugden langs veld,
        Zij razen // en blazen,
        En spelen op het Riet:
        het Is wonder // dat onder
        Hunlieden geschiet.
      
      
        Zij lopen, zij hopen
        Het Kindtje te zien;
        Zij willen // het stillen,
        Zij willen het verblien,
        Met kussen // met sussen,
        Met vrindelijckheyd,
        Met kallen // met rallen,
        Met Herders-beleid.
      
      
        Heus Kloris // en Doris.
        Waar loopt gij dus ras?
        Laat zoetjes // uwe voetjes
        Volbrengen haar pas.
        Wij moeten // gaan groeten
        Te samen den Heer;
        het Is reden // 'k wil mede
        Bewijzen Hem eer.
      
      
        O Tyter ,, wat Cyther,
        Wat snaare-gespel,
        Wat Orgel // wat gorgel
        Behouwen wij wel
        Om de ere // des Here
        (Die het herte verheughdt)
        Te spelen // te kwelen
        Met des herten geneughd.
      
      
        Trompetten // Kornetten
        Behouven wij niet,
        Noch Bullen: // wij zullen
        Van buiten een Lied
        Beginne // uit minne
        Van het Opperste Goed;
        Waar henen // onze benen
        Nu lopen met spoed.
      
      
        O Maetjens // ons praetjens
        Die dienen gestaeckt;
        Houdt Herder! // nier varder:
        Het Stal'ken genaakt.
        Laat vallen // ons allen
        Ter aarde ter neer.
        En groete // dit zoete
        Klein Kindeken teer.
      
      
        Zijt welkom, zijt welkom,
        Zijt welkom weerom
        Bestierder // Vercierder
        Van Sterren // en blom;
        Wien baarde // op de aarde
        Dit Maeghdeken fier,
        Wiens glory // memory
        Wij vieren nu hier.
      
      
        Komt Machten ,, komt Krachten
        Van het hemelse hof;
        Komt Eng'len ,, vermeng'len
        Uwe stemme tot lof
        Onze Broeders ,, onze Hoeders,
        Met hemelsch' jolijt:
        Zegt Amen: ,, weest te samen
        Met de Herders verblijd.
      
      
        Messia ,, Maria,
        Wij groeten U te zam;
        Verspreiden // verbreiden
        Wij zullen uwe Naam:
        Hier leggen // onze Weggen,
        Hier laten w' onze Melk;
        Wij jachten // wij trachten
        Na huis toe weer elk.
