
    
      
        V wel bekwame // Bruid,
        Door uwen Name // uit
        Helpt rechten wonder // dingen,
        In u macht, sterken // Held,
        Men 'svyandts werken // veld,
        En al kan te onder // bringhen.
      
      
        In uwe grote // macht,
        Eer ik de bloote // slacht,
        Helpt mij toerusten // stadig.
        Want tis van node // wel,
        Op dat ik dode // snel
        Altijdts de lusten // schadigh.
      
      
        T'gheloove t'eenen // schild
        Mij altijdts lenen // wilt,
        Want meer dan stalig // dezen
        Mij dient in elken // tijd,
        En ook in welken // strijdt,
        De hop' om zalig // wezen,
      
      
        Dien Hellem vryde // mij
        Ook aan mijn zijde // zij
        'sGheests zweert, de waarheid // mede,
        Tis enen goeden // Kreeft,
        Die rein ghemoeden // geeft,
        In zuiver klaarheid // vrede.
      
      
        In liefde vyerigh // wis
        Die goedertierig  // is,
        Wilt mij rechtzinnig // gorden,
        Mag zij mij blijven // bij,
        Ik zal beklijven // vrij,
        En overwinning // worden.
