
    
      
        Een dienaar, lichte klaar // Godes schaer
        Als getrouw, held // stelt,
        Menich jaar // heeft hij haar voorspelt,
        T'oordeel zwaar // hoe't zal zijn geveld,
        En vertrokken // met wat wapen.
        Dat die bokken // van de schapen,
        Ras en snel // totter hel // wreed en fel
        In den eewighen gloed // moet,
        T'sal zijn wel // lustig ende zoet,
        Die het bevel // van den Here doet.
      
      
        Hier na hoort // zijn geboord // sach men voort
        Anno vijftien // schien,
        Na het akkoord // metten vromen lien,
        Menich oord // heeft hij moeten vlien,
        Men bedreven // d'uytvaert klachtig,
        Als men schreven // tweentachtich,
        Doen hij smertich // had goedthertich // een en dertig
        Iaer dat God verlient // dient,
        Duizend partich // ende die ghetient,
        Mede wartich // met Christo verient.
      
      
        Hij blies den horen // om te sporen // Godes toren,
        Dien hij over t'lant // sant,
        Wel te voren // dreychde hij Godes hand,
        Snel verloren // ginckt aan elken kant,
        Want zijn reen // die met beven,
        Van het gemeen // volk nat leven
        Sou ghevaet // zijn met raad // is versmaad,
        En niet op gepast // vast
        Godes raad // is menig een last,
        Die zoet praat // is haar liever gast.
      
      
        Een goede strijd grote spijt // in zijn tijd,
        Heeft hij wel ghestreen // leen
        Klaar verblijd // vrymoedech ghetreen,
        Den loop wijt // totten einde reen,
        T'gheloof ghesont // in het benouwen
        Tot alder stond // vast behouwen,
        Dat met recht // zij ghesecht // trouwen knecht
        Nu zal zijn die kroon // t'loon,
        Niet en vecht // tegen deze troon,
        En voorlecht // t'volck veel woorden schoon.
      
      
        Laat doch af // dat gij kaf // ende draf
        Heerlijk voor goe weit // spreyt,
        Een ander straf // u nu toebereid,
        Denkt om t'graf // dat gij niet en schreyt,
        Als die Heer // komen zal bezoeken,
        Werk en leer // na luid der boeken,
        Dat eerzaam // uwen naam // na het betaam,
        Werd gevonden daar // klaar
        Te zijn bekwaam // als ghetrou dienaar,
        Met hoger faam // in Godes bewaar.
      
      
        Ernstig weest // ende leeft // na den Geest
        Deze spreuke klaar // naar,
        Aldermeest // dien hij was te zwaar,
        Ende prest // hem noch openbaar,
        Kwaad kwaad // seytmen in het leven,
        Twelck staat // Proverbi twintig schreven,
        Alst vergaan // van die baan // en gedaan
        Is, zo prijstment weer // zeer,
        Wien vermaan // eyscht dan daad bij leer
        Moet vergaan // ja ook zelf die Heer.
      
      
        Noemter een // in het gemeen // die met vreen
        In zijn tijd geleefd // heeft,
        Is er gheen // mij gelijk hier geeft,
        Zoekt gheen leen // niet langer en sneeft,
        Moyses leed // dreigen om te stenen
        Elk Profeet // moste dikwijls wenen,
        Waarom want // alle sant // door het land
        Men al den Profeet // weet
        Weg van kant // een ieder creet,
        Wie haar hand // boodt creech mee een smeet.
      
      
        Ere groot // strak men boot // na die dood
        Die men de ontgaan // traan,
        Menich stoot // heeftmens aangedaan
        Nu isset noot haar weg moetmen gaan,
        Och hoe groen // wie zal des gezellen,
        Schrikken doen // voor die pijn der Hellen,
        Want waar met // niemand bet // weet Gods Wet,
        Elk vermaand met spoed // zoet
        Haar ghebedt // ende werken goed,
        Als trompet // men ook weten moet.
      
      
        Roemt niet meer // van een leer // die ter neer
        Door u zelfs ghestort // wordt,
        Zoekt gheen eer // om dit leven cord,
        V begeer // tot het eewich port,
        Dat men u niet // gaat verwijten,
        Dat gij ziet // als d'ypocrijten,
        Gemanierd // dat gij viert // en versierd,
        Der gherechter lijk // dijk,
        En u tiert // als van deuchden rijk
        Bedroeft gheviert // wordt bevonden slijk.
      
      
        Toont doch niet // dat men u ziet // met verdriet,
        In des Werelds doen // spoen,
        Dat gij u biedt // sterk // met woorden koen,
        Ende vlied // t'werck // om smenschen zoen,
        Die wereld tiert // en heeft behagen,
        Ia zij viert // die Heilige dagen,
        Prijst haar leer // met begeer // ende eer,
        En die nu noch die daad // vaat,
        Ende weer // oprecht leven gaat,
        Drucktse neer // ende wordt gehaat.
      
      
        Bovwen gij // ook als zij // even blij,
        Den want in valscher schijn // blijn
        Spreket vrij // het is die medecijn,
        T'bleeck hoort mij // aan haar kwaad fenijn,
        Israel // die zo ginck leven
        Is rebel // gants verdreven,
        Wt haar land // t'alder sant // haar vijand
        Veel haar bloot // vergood // root
        Berooft verbrand // met een slag of stoot,
        Dat niemand // ooit sach meer der noot.
      
      
        Sulcken noot // hoewel groot // ende snood,
        Daar God Israhel // snel
        Mee beslot // grouwelijck en fel,
        Was maar bloot // enen schadu spel,
        T'ware wezen // is gelegen
        Zo wij lezen // Marcus negen,
        T'worm niet sterft // men beerft // en bederft,
        Enen eeuwig vuur // zuur,
        T'welck doorkerft // geest ende natuur,
        Niet meer verft // vreest t'eeuwich ghetruer.
      
      
        Och hoe veer // vlied dan eer // als men teer
        Al des werelds staat // laat,
        Aan den Heer // t'ontijt zoekt ghenaet,
        Och hoe zeer // weinig die het recht vaat,
        Onder die het // woord Gods dienden,
        Menich niet // die het recht minden,
        Immers gheen // hoort mijn reen // dan alleen
        Timotheus daar // waar
        Paulus gheween // ende nu ook zwaar
        Achtme cleen // nauwelijks een haar.
      
      
        Enen staat // ende raad // gelijke daad,
        Als vant begin ghesweeft // heeft,
        T'selfde maet // men als nu noch geeft,
        Even kwaad // als voor die mense leeft,
        Zo treffen necht // alle die Schriftueren
        Effen recht // tot dezer neren,
        Ia all' ellent // en torment // God haar sent
        Die nu zo onbedacht // lacht,
        Die hem ook went // uit alle zijn kracht
        Blijft ongeschend // want God hem wacht.
      
      
        Neemt in dank // dezen sanck // u verlanck
        Zij zo veel gij meucht // denkt,
        Poocht u ganck // d'outheyt als die jeugd,
        Om eer lank // t'erven d'eeuwighe vreugd,
        Krijgt gij iet // hier bedenken
        Aan dit lied // wie't mag schenken
        Ion bericht // geeft u licht // dit gedicht
        Noemt hem in dit vars // dwars,
        Al diet sticht // vast staat // klemt als hars
        In het kwaad zwicht // ontgaet // Satans pars.
