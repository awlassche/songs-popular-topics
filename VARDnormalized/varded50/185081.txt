
    
      
        Als Palmen groenen // zeer suyverlijc,
        En Cederen doenen // zeer heerlijk,
        Zo moet gij bloeien // in deugden rijk,
        Wassen en groeien // zeer lieffelijck,
        Op dat gij o jeugd // in ware deugd,
        Blij en verheugt // altijd recht lichten meugt.
      
      
        * Laat in u schynen // dat ware licht,
        En heel verdwijnen // dat u ontsticht,
        Zo moet gij bloeien // in deugden rijk,
        Wassen en groeien // zeer lieffelijck,
        Op dat gij o jeugd // in ware deugd,
        Blij en verheugt // altijd recht lichten meugt.
      
      
        Stelt u gelaten // in stilligheyt,
        En wilt aenvaten // goedwilligheid,
        Zo zal dan op dijn // vloeiende zijn,
        Godes Geest fijn // alst water uit den Rijn.
        Op dat gij o jeugd // in ware deugd,
        Blij en verheugt // altijd recht lichten meugt.
      
      
        * Zo zal dan vloeien // van u eerbaar,
        Stromen die vloeien // zeid die schrift klaar,
        Hoe zijnen gave // heeft die ontfaen,
        Geen rijcker grave // en vintmen zaan,
        Dan die mens rein // ootmoedig klein,
        Die zo alleyn // drinkt uit ‘s levens fontein.
      
      
        * O reine ziele // die zo gheplant
        Staat in priele // schoon en playsant,
        Wel rijk gezegend // zijt gij voorwaar,
        Want op u regent // voor ende naar,
        Van God zeer goed // uit genae zoet,
        Des geestes vloed // waar voor gij wassen moet.
      
      
        * O jeugd bekwame, in sulcken hof
        Wens ik u t’same // om daar Gods lof
        Recht te verbreiden // als spruiten jent,
        Vol vruchtbaerheyden // zeer zijn geënt,
        Aan die wijnrank // vol zoeten drank,
        Daar niet uit sprank // dan God tot lof en dank.
