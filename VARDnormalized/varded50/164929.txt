
    
      
        Ik zal voorwaar // den Hemelschen vaar,
        In het openbaar // gaan prijzen,
        Zijn lof zeer klaar // zal voor en naar,
        Bij zijn dienaar // noch rijzen,
        Die ziele mijn // zal met blijde schijn,
        Hem verheuge dijn // o wijze,
        Dat het hore fijn // op dit termijn,
        Die ellendige dijn // en verjolijse.
      
      
        O Goddelijke graan // met malkander beraen,
        Met druk belaen // verschoven,
        Laat ons gheraen // met Davids vermaan
        Des Heren naam // gaan loven,
        Doen ik den Heer // noch zochte van veer,
        Hij antwoorde mij weer // van boven,
        En verloste mij teer // door zijnen leer,
        Van alle mijn zeer // mij beroofde.
      
      
        Die hem aanziet // en tot hem vlied,
        Die en wordt niet // tot schanden,
        Want God die biedt // die ellendige met vliet
        Die tot hem riep // zijn handen,
        En helptse voort // al worden ze verstoort,
        Wt alle noot // en banden,
        Zijn engel hoort // beleghert met akkoord
        Der gheloovigher oord // spijt vijanden.
      
      
        Ziet ende smaakt //watten vriendelijcken sak
        Dat hem genaakt // na dezen,
        Die na God haeckt // en op hem waakt,
        Niet en versaeckt // maar vrezen,
        Vreest God altijd // en hem belijt,
        Gij die nu zijt // verrezen,
        Want die met vlijt // hem vreest en belijd,
        In dezen tijd // zal zijn geprezen.
      
      
        Werpt u begeer // allenig op den Heer,
        Voort kwaad ghebeer, // u wilt hoeden,
        Die hongerde zeer // na zijn Goddelijke eer,
        Vervulde hij weer // met goeden,
        Maar die hem versmaad // nu zijn verzaad
        Rijk hoog van staat // van moede,
        God weer verlaat // nootdruftich gaat,
        En hem genaakt veel teghenspoede.
      
      
        Hierom verlaat // dit vuyle gewaad
        En sulcken staat // mispresen,
        Gods vrede aenvaet // die s'menschen raad,
        Te boven gaat geprezen,
        En geeft gheen wijk // is het waarachtelijk,
        Oprecht cuysch lieffelijck // in het wezen,
        Is het wel luydelijck, is het deugdelijk,
        Is het lof eerlijk // gheresen,
      
      
        Komt herwaarts hoort // Gods kinderen zijn woord,
        Na waarheid sport // te degen,
        Wie is er voort // die na het behoort
        Met goed akkoord // te leven,
        En wie gerust // tot goede dagen heeft lust,
        Ion tonge uytblust // voor sneven,
        Ion lippen sust // van bedriegen onrust,
        En boze lust // te streven.
      
      
        Vant boos wilt vlien // doet goed nadien,
        Na vrede allien // wilt jagen,
        Want God ten tien // wil haar ansien,
        Die hem ghemien // behagen,
        Ook haar gebeen // en droevig gheween,
        Wilt de Here alleen // dragen,
        Die haar afscheen // en boosheid deen
        Wil dit Here van een // verjagen.
      
      
        Die Here voldoet // t'ghebedt zeer zoet,
        Eens verslagen ghemoet // vol smerten,
        En hij is goed // die tot hem spoed,
        Gebroken vroed // van herten,
        Die gherechticheydt // voor die mensen verbreydt,
        Van t'quaet hem scheid // met moeden,
        Hoewel hij schreyt // hem zijn bereid
        En toegezeid // veel goeden.
      
      
        Maar die nu lacht // en wordt geacht,
        Van s'menschen geslacht // geprezen,
        En toont zijn macht // over t'gheslacht
        Die uit al haar kracht // God vrezen,
        Die wil God weer // werpen ter neer,
        Zal worden zeer // begresen,
        Door Godes leer // die nimmermeer
        Zal nemen keer // is het bewezen.
      
      
        Hierom gij al // der Christenen getal,
        Treedt den weg smal // na t'leven,
        Al rister bal // druk ongeval,
        Hier in dit dal // der sneven,
        Voor t'cleyne verdriedt // d'welcken hier schiedt,
        En wilt doch niet // zo vrezen,
        Maar hier op ziet // wat God onthiet,
        En ons aenbiet // te geven.
      
      
        Hij roept komt bij // die beladen zij
        Ik zal u maken vrij // van zonden,
        Scheid uit het gecri // wilt wezen blij,
        Mijn volk zult gij // zijn bevonden,
        Mijn jok is zoet // dat op u doet
        Mijn last is goed // den gezonden,
        Maar t'is al roet // en tegenspoed,
        Die zijn verwoed // mijn gronden.
      
      
        O Princelijcke God // zijt gij ons deel en lodt,
        Bidden wij in slot // ons sanghen,
        Tegen s'werelts spot // door u gebod,
        Sterkt uwen rodt // haar gangen,
        Vwen geest zendt // dat wij present,
        T'gheloove jent // aanhangen,
        En s'Hemels tent // na dezen endt,
        Zeer excelent // ontvangen.
