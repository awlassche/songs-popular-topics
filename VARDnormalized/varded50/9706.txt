
    
      
        HEer uwen name // is groot eersame,
        Wijd uytghebreyt // is uwe fame,
        V Majesteit // vol heerlijkheid.
      
      
        Heer u genade // duert vroeg en spade,
        Zonder verganck // eeuwig gestade,
        Even zo lank // zij u lof dank.
      
      
        Here uwe wegen // zijn al te degen,
        Altijd tot goed // zijt gij genegen,
        Lof u zijn moet // in overvloed.
      
      
        Heer uwe werken // die doen ons merken
        Dat gij altijd // niet om versterken,
        Almachtig zijt // ghebenedijt.
      
      
        O Heer u wezen // moet zijn geprezen,
        V goedheid heeft // ons al genezen,
        O Heer gij geeft // daart al bij leeft.
      
      
        O Heer u woorden // ons met accoorden
        Wijzen hoe dat // wij al behoorden
        Te gaan den pad // na u schoon Stad.
      
      
        V hulp, o Here // hier te allen kere
        Behoeft ons wel // want wij zijn teere,
        Dus maakt u snel // ins ons bevel.
      
      
        Heer, uwen wille // vrij van gheschille,
        Leert ons al doen // van zonden stille
        Als ranken groen // op elk saysoen.
      
      
        O Heer, u Rijke // daar toe gelijke
        Ons weerdig maakt // dat niemand wikke
        Verloren naakt // maar neerstigh waakt.
