
    
      
        Ik arme Kind // waar zal ik keren
        Dan mij verneren // onder Gods hand,
        Dat mij den wind // der valscher leren
        Niet en mag deren // Heer doet bijstand.
      
      
        O Heer God Vader // vol van genade,
        Van alle kwade // mij doch bewaard,
        Gij Fontein-ader // vloeyt vroeg en spade
        Op dijnen zade // van diner aard.
      
      
        O God, doet mij // verjubileren,
        Ik zal landeeren // u Name groot,
        Al soud' men vrij // mij persequeren
        En tribuleren // tot in der dood.
      
      
        Zal ik niet haast // met u verblijden,
        In witter zijden // o Vader jent?
        Het leven verbaasd // is niet dan strijden
        Tot allen tijden // druk en ellendt.
      
      
        Tot u, o Here // wil ik mij geven,
        Mijn steen verheven // mijn toeverlaat,
        Want in u lere // vind' ik het leven,
        Hoe soud' ik beven // als gij bystaet.
      
      
        Ik bidd' u seere // om uwe gracy,
        Tot elker spacy // verhoord u Bruid,
        Vertroost u teere // door jubilacy,
        Als des Werelds nasi // zal roepen luid.
      
      
        Als haar de Mensen // zeer verblijden
        Iagen en rijden // nae's Werelds loop,
        Dan is mijn wensen // en mijn belijden,
        Wilt mij bevrijden // voor sulcken hoop.
      
      
        Na u allene // zijn mijne gangen,
        Ik ben behangen // met zonden snood,
        Na anders gene // staat mijn verlangen,
        Wilt mij ontvangen // Heer in der noot.
      
      
        Zal des turbacy // noch lange duiren?
        Dit doet mij truiren // troost mij, o Heer,
        Want door blamacy // zij veel vervuiren,
        Door het versuiren // in uwe leer.
      
      
        Zo wilt bewaren // sparen en wachten
        Mij door u krachten // mijn Middelaar,
        Wilt mij vergaren // leiden en achten,
        Bij de geslachten // onder het Outaer.
      
      
        O God mijn Herder, doet mij doch groeien
        Door liefde bloeien // als rank vruchtbaar,
        Ik weet geen weerder // wilt mij besproeien,
        Laat op mij vloeien // u Geest eerbaar.
      
      
        O gij Fonteine // vol goeds geprezen,
        Daar uit geresen // is het leven jent,
        Gij kont alleyne // ons Ziel genezen,
        Wilt bij ons wezen // tot in den ent.
      
      
        Nu bid' ik Prinse // mijn God almachtig
        Weest nu doch achtig // u Schaepkens al,
        O Heer, bemintse // uit liefden krachtig,
        Weest ons gedachtig // in het jammer-dal.
