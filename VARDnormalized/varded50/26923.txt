
    
      
        O Here God van machten // groot,
        Voor u zijn mijn gedachten // bloot,
        Daar en is niet zo ergen // man,
        Die hem voor u verbergen // kan,
        Want gij binnen en buiten // ziet,
        Men kan voor u opsluiten // niet.
      
      
        T'is voor u als den lichten // dag,
        Geeft in mij dat wel stichten // mag,
        Enen nieuwen bekwamen // geest,
        Met al die uwen Name // vreest,
        Dat ik mag hier beneden // plein,
        Wandelen in der vreden // rein.
      
      
        Zonder den vrede zeer dier // baar,
        En zal men mogen zien hier // naar,
        God in zijn eeuwig wezen // fijn,
        De knecht zal niet geprezen // zijn,
        Die met onvrede kwellen // gaat,
        En zijn medeghesellen // slaat.
      
      
        Met stokken oft met roede // stijf,
        En meen ik op dat goede // lijf,
        Maar met der tongen en haasten // praat,
        Spreken van zijnen Naasten // kwaad,
        Zo niet der tonge griefde // rat,
        Maar niet en leert de liefde // dat.
      
      
        Al trockt gij ook een bloedig // mes,
        Tegen u vriend die nodig // es,
        En u goed principalijck // riet
        Zo doet gij noch kwalijk // niet,
        Als dat gij met versmaden // maakt,
        Hoe men in kromme paden // raakt.
      
      
        Die zijnen mond tot laster // went,
        Den vred' hij veel t'onvaster // kent,
        Want naar Syrachs gewagen // goed,
        Hij doet van al zijn dagen // boet,
        Die hem tot kwalijk spreken // geeft,
        En selue vol gebreken // leeft.
      
      
        Here bewaard mij in de // pad,
        Naar die reine beminde // stad,
        Mijn tonge en mond benenen // dien,
        Dat ik hier naar mag t'leuen // zien,
        En goede dagen mede // voort,
        Och Here mijner bede // hoort.
      
      
        Mijn ghebedt, Heer, verhoren // wilt,
        Dat daar deur uwen toorn // stilt,
        Ouer alle mijn zonde // boos,
        Door u genade, gronde // loos,
        Vergeeft mij die al tezamen // vroed,
        Want ik mij voor u schamen // moet.
      
      
        Laat mij, Here voor u vri // gaan,
        Van zonden, als den Publi // caen,
        Die voor u in den Tempel // bat,
        Geeft mij naar dat exempel // wat,
        Dies gij naar u verklaren // vast,
        Kwijt maecktet van den zwaren // last.
      
      
        Of ik u zo behaagde // reen,
        Heer gelijk als Magde // leen,
        Doe zij verkoos dat beste // deel,
        Sold gij haar kwijt de reste // heel,
        Der zonden // tot verzoeten // dies,
        Zij, met tranen u voeten // wies.
      
      
        Veel biddens, zij met tranen de,
        Ook dat Vrouken van Canane
        Zij bad met goed betrouwen // snel,
        Voor haars dochters benouwen // fel,
        Ootmoedig en van herten // leech,
        Van u zij haar begeerden // creech.
      
      
        Haar dochter deed gij, Heer, bijstand,
        Zij was gekweld van den vijand,
        Deed' haar veel persequeren // an,
        Maar gij zacht dat verneren // van,
        S'moeders ghelooue machtig // sterk,
        Zo toondet gij u krachtig // werk.
      
      
        Zeer veel krachtige daden // zoet,
        Here, gij deur ghenaden // doet,
        Wie tot u met ghelooue // kwam,
        Melaatse, Kreupel, Dooue // Lam,
        Den Blinden sach, den Stommen // sprak,
        Gij hebt van haar genomen // t'pack.
      
      
        Och Here God barmhertig // Vaar,
        Den Man die achtendertich Iaer
        Op het Bedt hadde gelegen // krank,
        Hebt gij cloeckheyt en degen // ganck,
        Wt liefden hem goed rond // ghejont,
        En maecktet hem terstond // ghesont.
      
      
        Princelijcke God diuijn // zijt mijn
        Een eeuwig Medecijn // met wijn
        En oly mij inwendig // bint,
        Waar dat gij het, Heer, ellendig // vint,
        In mij een geest verslagen // schept,
        Daarin gij goed behagen // hebt.
