
    
      
        het GRootste vermakken // dat meest verblijd
        In kwade zaken // is korten tijd.
        het Meeste verblijden // het welk het goede geeft,
        Zijn lange tijden // dat men het heeft;
        Wanneer ik ter deeg // dit overweg,
        Denck ik; Gelukkig zijn wij om leegh.
      
      
        het Leven beneden // ons lastig wordt,
        Maar het is ook heden // geweldig kort,
        Schoon wij bezuren // hier veel verdriet,
        het Zal niet lang duren// het is haast te niet.
        Maar het leven vol eer // bij God den Heer
        Dat zal niet eindigen immermeer.
      
      
        Salig' ellenden // en ongeneught,
        Die altijd enden // in lust en vreugd,
        Zalig vermakken // dat niet en wordt
        Door enige zaken // ooit af-gekort,
        En zaligen end // dat ons versent,
        In vreugd, daar het einde gheen vreugde schent.
