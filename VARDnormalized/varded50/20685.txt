
    
      
        DOrathee ,, ach! uwe twee Zusters, God begeven.
        Zij bezoeken // (het wil niet loecken)
        Of Gij, g'lijck als zij, wilt leven.
        Zij bezoeken.
      
      
        Maar gehecht // en gevlecht
        Zijn met liefdes-banden
        All' uwe zinnen // all' uwe minne,
        Aan den Heer van alle landen.
      
      
        G' hebt geleerd // G' hebt bekeert
        Uwe Zusters deftig;
        En Zij leden // en Zij streden
        Voor de lering Christi treftig.
      
      
        Grote nood // ja de dood
        Hebben Zij verdragen:
        Hoge gaven // des Hemels haven
        Kregen Zij na deze vlagen.
      
      
        Doen zijt Gij // in de lij
        We'er daarom gekomen:
        In die dagen // hebt Gij slagen
        Lustig op uwe le'en vernomen.
      
      
        Ei! Gij hebt // en Gij schept
        Vrolijkheid in het lijden.
        In het blaken // in het krakken
        Uwer le'en, Gij U verblijden.
      
      
        Gij ginght heen // Gij ginght tre'en
        Na het schavot blijmoedig.
        U verbeiden // des Hemels weiden
        En Gij kwaamt daar in gans bloedig.
      
      
        Uit dat land // heeft een mand'
        God met schone Rozen
        Ne'er gezonden // op dier stonden,
        Die Theophilus zag blozen.
      
      
        Deze man // nam die an,
        En Hij zich bekeerden;
        Ja zijn leven // heeft gegeven
        Voor den Heer des Rechtsgeleerden.
      
      
        des Hemels lot // schonk hem God
        Voor des minne-voncken;
        Om zijn deughden // is van vreugden
        Hij, en Gij, gestadig dronken.
      
      
        Och of Gij // ook op mij,
        Dorothee, woud denken!
        Ei! bidt krachtig // God Almachtig,
        Dat Hij mij zijn Hof wilt schenken.
