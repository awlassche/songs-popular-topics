
    
      
        MEt Brood // met Bier // zeer fijn met Wijn,
        Pastei // Lamprei // met Haas en Konijn,
        Met Hoen // Kapoen // Wildbraad // verzaad
        Elk is // gewis // op dit termijn.
      
      
        Ons hert is heel met vreugd // verheugt
        Ons moet // zeer zoet // strekt tot gheneucht
        Wij zingen // en springen // wij klinken en winken
        Versterkt o Heer ons jeugd // en denkt.
      
      
        Men ziet hier staan // ghesoon ghebraen
        Die spart // die Taart // die is belaen
        Mersepeyn // met klein // begint het hint
        Die haan // en zwaan // kan niemand schaen,
      
      
        Wij danken u veel meer // o Heer
        Dan wij // zeer blij // u schaepkens teer
        met monden // verkonden // versterkt en werkt
        In ons nu voort // u woord en leer.
