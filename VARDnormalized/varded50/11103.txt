
    
      
        CVpido cleen// wie geeft u sulcken macht
        Dat gij alleen// benemen moght mijn kracht,
        V booghsken fijn// ras op een ander stelt,
        En laat mij zijn// in vrijheid onghequelt.
      
      
        Komt gij in het veld// met sulcken overmoed,
        Het is geweld// dat gij de jonkheid doet,
        Die u bemint// die doet gij meest verdriet,
        Al heet gij blind// mij dunkt nochtans gij ziet
      
      
        Wie u ontvliet// die seynt gij enen schicht,
        Dat hij te niet// moet gaan doort schoon gezicht
        Van een rein maagd// die hij van herten vrijt,
        Daar gij hem jaagt// door uwen pijl altijd.
      
      
        Wilt gij met nijt// u t'onswaerts tonen wreed
        Zo wordy kwijt// u dienaars door dit leed,
        Men zal u meer// niet noemen minnaers Got,
        Gij zout dan zeer// van elken zijn bespot.
      
      
        Die u ghebot// nu volgen na't kompas
        Die zouden 'tlot// dan anders worpen ras,
        Ten is niet goed// te hebben sulcken hoofd,
        Die gants verdoet// zijn dienaars en berooft.
      
      
        Hierom gelooft// gij Prins, gij God van min
        Zo gij verdooft// noch eens mijn liefkens zin
        En zo zij fel// niet horen wilt mijn klacht,
        Zo keer ik snel// van u met gantser macht.
