
    
      
        GEbeden // seyndt heden // dijn reden // tot God,
        Om zwijgen // van het hijgen // dat krijgen // verdwenen,
        Moord, branden // ons’ landen // met schanden // bespot,
        Daar lusten // ons kusten // in rusten // voor-henen;
        Wilt spenen // o Here! het vermeerende kwaad,
        Ons zonden // ontbonden // dijn wonden // tot smaad,
        Maar behoed ons // en doet ons ghenaed’.
      
      
        Wilt sterken // de werken // met merckende reen
        Der staten // die het kwaad en // het haten mag scheuren,
        Verlichtse // en stichtse // ja vlechtse in een,
        Dat sy-lien // en wy-lien // ons vrij zien // van treuren:
        De deuren // wilt sluiten // van buit ende roof,
        Bevrijd // voor den strijdt // en den nijd // het Geloof,
        Maakt voor stoorders // de hoorders doch doof.
      
      
        Wij stromen // in dromen // door het schromen // verblindt,
        Noch zoeken // noch boeken // verkloeken de zinnen,
        Ons machten // ons grachten // ons krachten // zijn wind,
        God stuert // en bemuert // ons gebuerten van binnen;
        Gewinnen // ons voeten // t’ontmoeten // der deugd,
        Wij kleven // aan het leven // verheven in vreugd,
        Dat altijd // ons bevrijd, en verheugt.
      
      
        O hogen // vermogen // mee-dogenden Al!
        Dijn handen // verpanden // de banden // van vreden,
        Worpt neder // ‘het onweder // dat seder // het geval
        Onspoedig // verwoedig // ons bloedig vertreden:
        Dijn leden // beschermt // en om-ermt // uwe Bruid,
        Die het smachten // heel nachten // in klachten // weent uit,
        Laat doch broeien // d’on-bloeyende Spruit.
      
      
        De wolken // vertolken // de volken // den Peys,
        het Verhopen // staat open // om kopen genade;
        het Betrouwen // drijft rouw’ en // het benouwen op reis;
        het Verschillen // der willen // zal stillen // ter stade;
        Te spade // het geluk // in den druk // nooit verscheen:
        Bidt krachtig // eendrachtig // het waarachtige Een
        Zal ons geven // te leven // in Vreen.
