
    
      
        WAnneer men qua geruchten // hoort,
        Van vluchten // voort,
        Is het dan al veel met zuchten // 'twoort
        En duchten // stort
        Het goed // ghemoet // van binnen // maar
        Te vroeg zijn ons zinnen // zwaar,
        Want 'tkan wel wezen,
        Dat niet zo groot // en is den noot,
        Als vrees in dezen.
      
      
        Die qua maren beseven // zij,
        Doen beven // mij,
        Sechtmen, waar zal men leven // bij,
        Met weven // wij
        En spinnen // winnen // kwalijk // geld,
        Dat ons principalijck // quaelt,
        Dus in het gemene,
        Is grote diert' // ongemanierd,
        En nering kleene.
      
      
        Doch neeringhe van lijnen // mank
        In zijnen // ganck,
        Kan somtijds om verdwijnen // lank,
        Wel schynen // krank,
        Maar zaan // vergaan // noch sterven // ziet,
        Men mag het lijnen derven // niet,
        In 'swerelts ronde,
        Ten waar dat men // noch kin noch ken,
        Daar in en vonde.
      
      
        Voor al in dit verseren // met
        Des Heren // Wet,
        Moeten wij lijdtsaem leren // bet,
        Afkeren // het
        Al het onverduldig wroegen // fel,
        Laten ons genoegen // wel,
        Met dat voorhanden
        Is, hoe benouwt // die God betrouwt,
        Komt niet tot schanden.
      
      
        Laat hopen om genezen // zoet,
        Op dezen // voet,
        Is het kwaad, het kan haast wezen // goed,
        Gheresen // spoed,
        Gelijk den tijd zijn gangen // gaat,
        Wij moeten ontvangen // kwaad
        In danke mede,
        Ghylijck als Iob // zijn antwoord' op
        'tWijfs reden dede.
      
      
        Prinsen die met den kwaden // tyt
        Beladen // zijt,
        Laat u Gods Woord beraden // wijt,
        De schaden // kwijt,
        Werd gij, ja al het bezwaren // van
        De felle kwade maren // dan,
        Die somtijds rijzen,
        Niet is zo goed // dan wat God doet,
        Van als hem prijzen.
