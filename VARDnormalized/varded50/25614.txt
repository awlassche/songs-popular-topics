
    
      
        Dancklijck eenmoedig // prijst God den Vaar,
        Want hij zeer goedig // is voorwaar,
        Zijn  gonst om prijzen // gaat hij eenpaar
        Ons nu bewijzen // in het openbaar,
        Och vrienden gij //  zingt doch met mij
        Van herten blij // dankt doch God allegaar
      
      
        In het huis des Heren // zullen bereid,
        Wij gaan met eren // zo hij toeseyt,
        Om vreuchd’ verwinnen // hier boven gaan,
        Ons voeren binnen // de poorten staan,
        Blijft op den ganck // u leven lank,
        Al zijn wij krank // die hop’ v’steect ons zaan
      
      
        Een stad ghebouwen // heerlijk en breyt
        Haar Borghers houwen // die  enigheid,
        Men mag bij trouwen //  zeer hoog gesteld
        Haar licht aanschouwen // het lucht in het veld
        Och vrienden jent // maakt ook bekent,
        Dat gij nu bent //  borgers elk als een held.
      
      
        Recht doch u wezen // als Christen goed,
        Aan Ziel geprezen // hert en ghemoet,
        Als de eerste deden // met Ziel en lijf,
        Enig in vreden // te samen stijf,
        Gods glory schoon // spant daar de Kroon
        In s’Hemels Troon // zal blinken haar bedrijf.
      
      
        In deze mueren, eert God den Heer,
        Lieft zonder trueren // elckandren meer,
        Gaat opwaarts beden // voor God aldaar,
        Drijmael t’gheeft reden // in het nieuwe Iaer,
        Hem offer bringht // dankt, loeft en zingt,
        Dit vier ontsinckt // de liefde in ons klaar
      
      
        Klimt op de graden // volheid der denkt,
        Dat gij geraden // t’gheloove meucht
        Betuigen waarlijk // als volk Gods vroed
        Wt God schijnbaerlijck // gesproken zoet,
        Met kracht en daad // zo Christus raad,
        Elk huis bestaat // dat men zo bouwen doet
      
      
        Kent, wilt gevoelen // daar zijn gezet
        De Rechter-stoelen // na rechter Wet
        Oordeel te geven //  in Davids huis.
        Na een elks leven // zonder abuis
        Krijgt elken slecht // daar loon na recht,
        D’onnutte knecht // zal daar worden confuys
      
      
        Perfeckten vrede // soo’t God behaagt,
        Krijgt niemand mede // dan die nu draagt
        Liefde te voren // in het herte hier bij,
        Rein uit verkoren // zonder  partij,
        Godlozen ziet // en hebbens niet,
        Want vrede diet // Christus den zijnen vrij
      
      
        Ik wens u tezamen // vrede van God,
        Liefde met namen // t’nieuwe gebod,
        Naar Schrifts verklaren // Ierusalem
        Wilt openbaren // volgt Christi stem,
        In denkt gaat voort // al na zijn woord,
        Niet anders hoort // en zal behagen hem.
      
      
        En in u torens // breete ende smal
        Moet zijn te vorens, vred’ over al,
        Daar men behoedig // de wapens hout,
        Bereid en spoedig // zeer menichfout,
        Om de trompet // te blazen net,
        Zijt gij gezet // t’volck Gods is betrout.
      
      
        Treedt op u voeten // t’gheluydt nu staat,
        Ontwaken moeten // zij niet te laat,
        Den vred’ wilt preken //  straft nu bij tijd,
        Zo onbezweken // u Ziel bevryt,
        Gods volk certein // de kudde klein
        Bewaard nu rein // want gij haar wachter zijt.
      
      
        En  wassen, groeien // moet gij Gods stad
        En voerich bloeien // te wens u dat,
        Om t’lof te bringhen // voor Gods aanschijn,
        Is nu het zingen // en t’wenschen mijn
        Vreedlijck vergaard //  Godtlijck gheaert,
        Zo Schrift verklaart // met Christen zinnen fijn.
      
      
        Recht op’t vervallen, bindt met der hand,
        Heer, onder allen, des vredens band,
        Bouwen, vergaren // zouden wij sterk,
        Als trou dienaren // zo aan u Kerk,
        Iae als een man // timmeren dan,
        Gheen vijand kan // ons dan beletten t’wercl.
      
      
        Staat om verhogen // Heer bij met kracht,
        Want onvermogen // is ons macht,
        Geenszins verblijden // en mogen wij,
        Helpt gij hier strijden // aan onze zij,
        In u macht staat // ons kracht en daad,
        Verwinnen laat // ons doch niet moede blij.
      
      
        Zaligheid nasi // gij Christen schoon,
        Zij u, zoekt stacy // in s’Hemels troon,
        Dat gij met namen // zonder bezwijk,
        Zo meucht verzamen // in het Hemelrijk,
        Liefdens vermaan // heeft dit gedaan,
        Neemt het doch aan // van mij zo al gelijk.
