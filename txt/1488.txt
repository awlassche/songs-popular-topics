
    
      
        MEt Broodt // met Bier // seer fijn met Wijn,
        Pastey // Lamprey // met Haes en Conijn,
        Met Hoen // Capoen // Wiltbraet // versaet
        Elck is // ghewis // op dit termijn.
      
      
        Ons hert is heel met vreucht // verheucht
        Ons moet // seer soet // streckt tot gheneucht
        Wy singhen // en springhen // wy clincken en wincken
        Versterckt o Heer ons jeucht // en deucht.
      
      
        Men siet hier staen // ghesoon ghebraen
        Die spaert // die Taert // die is belaen
        Mersepeyn // met cleyn // begint het hint
        Die haen // en swaen // can niemant schaen,
      
      
        Wy dancken u veel meer // o Heer
        Dan wy // seer bly // u schaepkens teer
        met monden // verconden // versterckt en werct
        In ons nu voort // u woort en leer.
